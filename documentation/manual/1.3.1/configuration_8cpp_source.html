140 void ConfigParser::read(const std::string &filename, bool append)
141 {
142  // Macro to check API for match with the DLL we are using
143  LIBXML_TEST_VERSION
144 
145  // Read the XML file
146  xmlDoc *doc = xmlReadFile(filename.c_str(), NULL, 0);
147  if (doc == nullptr) {
148  throw std::runtime_error("Could not parse configuration file \"" + filename + "\"");
149  }
150 
151  // Get the root element
152  xmlNode * rootElement = xmlDocGetRootElement(doc);
153 
154  // check if the root name is the requeste one
155  std::string rootName(reinterpret_cast<const char*>(rootElement->name));
156  if (rootName != m_root) {
157  throw std::runtime_error("The name of the root element is not \"" + rootName + "\"");
158  }
159 
160  // Check if the version is supported
161  const xmlChar *versionAttributeName = reinterpret_cast<const xmlChar *>("version");
162  if (m_checkVersion && xmlHasProp(rootElement, versionAttributeName)) {
163  xmlChar *versionValue = xmlGetProp(rootElement, versionAttributeName);
164  std::string versionString((char *) versionValue);
165  xmlFree(versionValue);
166 
167  int version;
168  std::istringstream(versionString) >> version;
169 
170  if (version != m_version) {
171  throw std::runtime_error("The version ofthe file is not not \"" + std::to_string(m_version) + "\"");
172  }
173  }
174 
175  // Load the options in the configuration file
176  if (!append) {
177  clear();
178  }
179  config::XML::readNode(rootElement->children, this);
180 
181  // Clean-up
182  xmlFreeDoc(doc);
183  xmlCleanupParser();
184 }

191 void ConfigParser::write(const std::string &filename) const
192 {
193  int status;
194 
195  // Create a new XmlWriter for DOM tree, with no compression
196  xmlTextWriterPtr writer = xmlNewTextWriterFilename(filename.c_str(), 0);
197  if (writer == NULL) {
198  throw std::runtime_error("Error creating the xml writer");
199  }
200 
201  xmlTextWriterSetIndent(writer, 1);
202 
203  // Start the document
204  status = xmlTextWriterStartDocument(writer, NULL, config::XML::DEFAULT_ENCODING.c_str(), NULL);
205  if (status < 0) {
206  throw std::runtime_error("Error at xmlTextWriterStartDocument");
207  }
208 
209  // Start the root element
210  xmlChar *elementName = config::XML::encodeString(m_root, config::XML::DEFAULT_ENCODING);
211  status = xmlTextWriterStartElement(writer, BAD_CAST elementName);
212  if (status < 0) {
213  throw std::runtime_error("Error at xmlTextWriterStartElement");
214  }
215 
216  // Add an attribute with version
217  std::ostringstream versionStream;
218  versionStream << m_version;
219 
220  xmlChar *versionAttr = config::XML::encodeString(versionStream.str(), config::XML::DEFAULT_ENCODING);
221  status = xmlTextWriterWriteAttribute(writer, BAD_CAST "version", BAD_CAST versionAttr);
222  if (status < 0) {
223  throw std::runtime_error("Error at xmlTextWriterWriteAttribute");
224  }
225 
226  // Write the configuration
227  config::XML::writeNode(writer, this, config::XML::DEFAULT_ENCODING);
228 
229  // End section
230  status = xmlTextWriterEndElement(writer);
231  if (status < 0) {
232  throw std::runtime_error("Error at xmlTextWriterEndElement");
233  }
234 
235  // Close the document
236  status = xmlTextWriterEndDocument(writer);
237  if (status < 0) {
238  throw std::runtime_error("Error at xmlTextWriterEndDocument");
239  }
240 
241  // Write the XML
242  xmlFreeTextWriter(writer);
243 }
