175 void SurfUnstructured::_dump(std::ostream &stream)
176 {
177 #if BITPIT_ENABLE_MPI==1
178  // Dump works only for serial calculations
179  if (getProcessorCount() != 1) {
180  throw std::runtime_error ("Dump of surfunstructured is implemented only for serial calculations.");
181  }
182 #endif
183 
184  // Space dimension
185  IO::binary::write(stream, getSpaceDimension());
186 
187  // Save the vertices
188  IO::binary::write(stream, getVertexCount());
189 
190  for (const Vertex &vertex : m_vertices) {
191  IO::binary::write(stream, vertex.getId());
192 
193  std::array<double, 3> coords = vertex.getCoords();
194  IO::binary::write(stream, coords[0]);
195  IO::binary::write(stream, coords[1]);
196  IO::binary::write(stream, coords[2]);
197  }
198 
199  // Save the cells
200  IO::binary::write(stream, getInternalCount());
201  IO::binary::write(stream, getGhostCount());
202 
203  for (const Cell &cell: m_cells) {
204  const ElementInfo &cellInfo = cell.getInfo();
205 
206  IO::binary::write(stream, cell.getId());
207  IO::binary::write(stream, cell.getPID());
208  IO::binary::write(stream, cellInfo.type);
209 
210  const long *cellConnect = cell.getConnect();
211  for (int i = 0; i < cellInfo.nVertices; ++i) {
212  IO::binary::write(stream, cellConnect[i]);
213  }
214  }
215 }

222 void SurfUnstructured::_restore(std::istream &stream)
223 {
224 #if BITPIT_ENABLE_MPI==1
225  // Restore works only for serial calculations
226  if (getProcessorCount() != 1) {
227  throw std::runtime_error ("Restore of surfunstructured is implemented only for serial calculations.");
228  }
229 #endif
230 
231  // Space dimension
232  int spaceDimension;
233  IO::binary::read(stream, spaceDimension);
234  setSpaceDimension(spaceDimension);
235 
236  // Restore the vertices
237  long nVertices;
238  IO::binary::read(stream, nVertices);
239 
240  reserveVertices(nVertices);
241  for (long i = 0; i < nVertices; ++i) {
242  long id;
243  IO::binary::read(stream, id);
244 
245  std::array<double, 3> coords;
246  IO::binary::read(stream, coords[0]);
247  IO::binary::read(stream, coords[1]);
248  IO::binary::read(stream, coords[2]);
249 
250  addVertex(coords, id);
251  }
252 
253  // Restore the cells
254  long nInternals;
255  IO::binary::read(stream, nInternals);
256 
257  long nGhosts;
258  IO::binary::read(stream, nGhosts);
259 
260  long nCells = nInternals + nGhosts;
261 
262  reserveCells(nCells);
263  for (long i = 0; i < nCells; ++i) {
264  long id;
265  IO::binary::read(stream, id);
266 
267  int PID;
268  IO::binary::read(stream, PID);
269 
270  ElementInfo::Type type;
271  IO::binary::read(stream, type);
272  const ElementInfo &cellInfo = ElementInfo::getElementInfo(type);
273 
274  int nCellVertices = cellInfo.nVertices;
275  std::vector<long> connect(nCellVertices, Vertex::NULL_ID);
276  for (int k = 0; k < nCellVertices; ++k) {
277  IO::binary::read(stream, connect[k]);
278  }
279 
280  CellIterator cellIterator = addCell(type, true, connect, id);
281  cellIterator->setPID(PID);
282  }
283 
284  // Build ghost exchange data
285 #if BITPIT_ENABLE_MPI==1
286  if (getProcessorCount()) {
287  buildGhostExchangeData();
288  }
289 #endif
290 }

322 void SurfUnstructured::extractEdgeNetwork(SurfUnstructured &net)
323 {
324  // ====================================================================== //
325  // VARIABLES DECLARATION //
326  // ====================================================================== //
327 
328  // Local variables
329  bool check;
330  int n_faces, n_adj, n_vert;
331  long id;
332  vector<int> face_loc_connect;
333  vector<long> face_connect;
334 
335  // Counters
336  int i, j;
337  vector<int>::const_iterator i_;
338  vector<long>::iterator j_;
339  VertexIterator v_, ve_ = vertexEnd();
340  CellIterator c_, ce_ = cellEnd();
341 
342  // ====================================================================== //
343  // INITIALIZE DATA STRUCTURE //
344  // ====================================================================== //
345  net.reserveCells(net.getCellCount() + countFaces());
346  net.reserveVertices(net.getVertexCount() + getVertexCount());
347 
348  // ====================================================================== //
349  // ADD VERTEX TO net //
350  // ====================================================================== //
351  for (v_ = vertexBegin(); v_ != ve_; ++v_) {
352  net.addVertex(v_->getCoords(), v_->getId());
353  } //next v_
354 
355  // ====================================================================== //
356  // ADD EDGES //
357  // ====================================================================== //
358  for (c_ = cellBegin(); c_ != ce_; ++c_) {
359  id = c_->getId();
360  n_faces = c_->getFaceCount();
361  for (i = 0; i < n_faces; ++i) {
362  check = true;
363  n_adj = c_->getAdjacencyCount(i);
364  for (j = 0; j < n_adj; ++j) {
365  check = check && (id > c_->getAdjacency(i, j));
366  } //next j
367  if (check) {
368  face_loc_connect = c_->getFaceLocalConnect(i);
369  n_vert = face_loc_connect.size();
370  face_connect.resize(n_vert);
371  j_ = face_connect.begin();
372  for (i_ = face_loc_connect.cbegin(); i_ != face_loc_connect.cend(); ++i_) {
373  *j_ = c_->getVertex(*i_);
374  ++j_;
375  } //next i_
376  net.addCell(c_->getFaceType(i), true, face_connect);
377  }
378  } //next i
379  } //next c_
380 
381  return;
382 }

457 unsigned short SurfUnstructured::importSTL(STLObj &STL, int PIDOffset, bool PIDSquash)
458 {
459  // ====================================================================== //
460  // OPEN STL FILE //
461  // ====================================================================== //
462  STL.open("in");
463  if (STL.err != 0) {
464  return STL.err;
465  }
466 
467  // ====================================================================== //
468  // LOAD ALL SOLID FROM THE STL FILE //
469  // ====================================================================== //
470  int pid = PIDOffset;
471  if (!PIDSquash) {
472  --pid;
473  }
474 
475  while (true) {
476  // ====================================================================== //
477  // LOAD SOLID FROM THE STL FILE //
478  // ====================================================================== //
479  int nVertex = 0;
480  int nSimplex = 0;
481  std::vector<std::array<double, 3>> vertexList;
482  std::vector<std::array<double, 3>> normalList;
483  std::vector<std::array<int, 3>> connectivityList;
484 
485  STL.loadSolid(nVertex, nSimplex, vertexList, normalList, connectivityList);
486  if (nVertex == 0) {
487  break;
488  }
489 
490  // ====================================================================== //
491  // PID OF THE SOLID //
492  // ====================================================================== //
493  if (!PIDSquash) {
494  ++pid;
495  }
496 
497  // ====================================================================== //
498  // PREPARE MESH FOR DATA IMPORT //
499  // ====================================================================== //
500  reserveVertices(getVertexCount() + nVertex);
501  reserveCells(m_nInternals + m_nGhosts + nSimplex);
502 
503  // ====================================================================== //
504  // ADD VERTICES TO MESH //
505  // ====================================================================== //
506  vector<array<double, 3>>::const_iterator v_, ve_;
507 
508  std::vector<long> vertexMap;
509  vertexMap.reserve(nVertex);
510 
511  long v_counter = 0;
512  ve_ = vertexList.cend();
513  for (v_ = vertexList.cbegin(); v_ != ve_; ++v_) {
514  VertexIterator i_ = addVertex(*v_);
515  vertexMap[v_counter] = i_->getId();
516  ++v_counter;
517  } //next v_
518 
519  // ====================================================================== //
520  // ADD CELLS TO MESH //
521  // ====================================================================== //
522  vector<array<int,3>>::const_iterator c_, ce_;
523  array<int,3>::const_iterator w_, we_;
524 
525  ce_ = connectivityList.cend();
526  for (c_ = connectivityList.cbegin(); c_ != ce_; ++c_) {
527  // Remap STL connectivity
528  int n_v = c_->size();
529  std::vector<long> connect(n_v, Vertex::NULL_ID);
530  we_ = c_->cend();
531  int i = 0;
532  for (w_ = c_->cbegin(); w_ < we_; ++w_) {
533  connect[i] = vertexMap[*w_];
534  ++i;
535  } //next w_
536 
537  // Add cell
538  CellIterator cellIterator = addCell(getSTLFacetType(n_v), true, connect);
539  cellIterator->setPID(pid);
540  } //next c_
541 
542  // ====================================================================== //
543  // Multi-body STL files are supported only in ASCII mode //
544  // ====================================================================== //
545  if (STL.stl_type) {
546  break;
547  }
548  }
549 
550  // ====================================================================== //
551  // CLOSE STL FILE //
552  // ====================================================================== //
553  STL.close("in");
554 
555  return 0;
556 }

573 unsigned short SurfUnstructured::exportSTL(const string &stl_name, const bool &isBinary, bool exportInternalsOnly)
574 {
575  // ====================================================================== //
576  // VARIABLES DECLARATION //
577  // ====================================================================== //
578 
579  // Local variables
580  int nVertex;
581  int nSimplex;
582  vector<array<double, 3>> vertexList;
583  vector<array<double, 3>> normalList;
584  vector<array<int,3>> connectivityList;
585  unordered_map<long, long> vertexMap;
586  array<int,3> dummyIntArray;
587 
588  // Counters
589  int v_count ,j;
590  vector<array<double, 3>>::iterator i_;
591  vector<array<int,3>>::iterator j_;
592  array<int,3>::iterator k_, ke_;
593  VertexIterator v_, ve_;
594  CellIterator c_, cb_, ce_;
595 
596  // ====================================================================== //
597  // INITIALIZE DATA STRUCTURE //
598  // ====================================================================== //
599  dummyIntArray.fill(0) ;
600 
601  nSimplex = m_nInternals;
602  if (!exportInternalsOnly) nSimplex += m_nGhosts;
603  vertexList.resize(getVertexCount());
604  normalList.resize(nSimplex);
605  connectivityList.resize(nSimplex, dummyIntArray);
606 
607  // ====================================================================== //
608  // CREATE VERTEX LIST //
609  // ====================================================================== //
610  i_ = vertexList.begin();
611  ve_ = vertexEnd();
612  v_count = 0;
613  for (v_ = vertexBegin(); v_ != ve_; ++v_) {
614 
615  // Store vertex coordinates
616  *i_ = v_->getCoords();
617  vertexMap[v_->getId()] = v_count;
618 
619  // Update counters
620  ++v_count;
621  ++i_;
622 
623  } //next v_
624  nVertex = getVertexCount();
625 
626  // ====================================================================== //
627  // CREATE CONNECTIVITY //
628  // ====================================================================== //
629  if (exportInternalsOnly) {
630  cb_ = internalBegin();
631  ce_ = internalEnd();
632  }
633  else {
634  cb_ = cellBegin();
635  ce_ = cellEnd();
636  }
637  i_ = normalList.begin();
638  j_ = connectivityList.begin();
639  for (c_ = cb_; c_ != ce_; ++c_) {
640 
641  // Build normals
642  *i_ = std::move(evalFacetNormal(c_->getId()));
643 
644  // Build connectivity
645  ke_ = j_->end();
646  j = 0;
647  for (k_ = j_->begin(); k_ != ke_; ++k_) {
648  *k_ = vertexMap[c_->getVertex(j)];
649  ++j;
650  } //next k_
651 
652  // Update counters
653  ++j_;
654  ++i_;
655  } //next c_
656 
657  // ====================================================================== //
658  // EXPORT STL DATA //
659  // ====================================================================== //
660  STLObj STL(stl_name, isBinary);
661  STL.save("", nVertex, nSimplex, vertexList, normalList, connectivityList);
662 
663  return 0;
664 }

706 unsigned short SurfUnstructured::importDGF(const string &dgf_name, int PIDOffset, bool PIDSquash)
707 {
708  // ====================================================================== //
709  // VARIABLES DECLARATION //
710  // ====================================================================== //
711 
712  // Local variables
713  DGFObj dgf_in(dgf_name);
714  int nV = 0, nS = 0;
715  long vcount, idx;
716  std::vector<std::array<double, 3>> vertex_list;
717  std::vector<std::vector<int>> simplex_list;
718  std::vector<int> simplex_PID;
719  std::vector<long> vertex_map;
720  std::vector<long> connect;
721 
722  // Counters
723  std::vector<std::array<double, 3>>::const_iterator v_, ve_;
724  std::vector<std::vector<int>>::iterator c_, ce_;
725  std::vector<int>::iterator i_, ie_;
726  std::vector<long>::iterator j_, je_;
727 
728  // ====================================================================== //
729  // IMPORT DATA //
730  // ====================================================================== //
731 
732  // Read vertices and cells from DGF file
733  dgf_in.load(nV, nS, vertex_list, simplex_list, simplex_PID);
734 
735  // Add vertices
736  ve_ = vertex_list.cend();
737  vcount = 0;
738  vertex_map.resize(nV);
739  for (v_ = vertex_list.cbegin(); v_ != ve_; ++v_) {
740  idx = addVertex(*v_)->getId();
741  vertex_map[vcount] = idx;
742  ++vcount;
743  } //next v_
744 
745  // Update connectivity infos
746  ce_ = simplex_list.end();
747  for (c_ = simplex_list.begin(); c_ != ce_; ++c_) {
748  ie_ = c_->end();
749  for (i_ = c_->begin(); i_ != ie_; ++i_) {
750  *i_ = vertex_map[*i_];
751  } //next i_
752  } //next c_
753 
754  // Add cells
755  int k;
756  for (c_ = simplex_list.begin(), k = 0; c_ != ce_; ++c_, ++k) {
757  // Create cell
758  i_ = c_->begin();
759  connect.resize(c_->size(), Vertex::NULL_ID);
760  je_ = connect.end();
761  for (j_ = connect.begin(); j_ != je_; ++j_) {
762  *j_ = *i_;
763  ++i_;
764  } //next j_
765  CellIterator cellIterator = addCell(getDGFFacetType(c_->size()), true, connect);
766 
767  // Set cell PID
768  int cellPID = PIDOffset;
769  if (!PIDSquash) {
770  cellPID += simplex_PID[k];
771  }
772 
773  cellIterator->setPID(cellPID);
774  } //next c_
775 
776  return 0;
777 }
