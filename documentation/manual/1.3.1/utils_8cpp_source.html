72 // ========================================================================== //
73 // INPUT //
74 // ========================================================================== //

76 // - m : int, upper bound of extraction interval //
77 // - list : ivector1D, list of extracted values //
78 // ========================================================================== //
79 // OUTPUT //
80 // ========================================================================== //
81 // - none //
82 // ========================================================================== //
83 
84 // ========================================================================== //
85 // VARIABLES DECLARATION //
86 // ========================================================================== //
87 
88 // Local variables
89 int N;
90 std::vector<int> set(m+1, -1);
91 
92 // Counters
93 int i, index;
94 
95 // ========================================================================== //
96 // INITIALIZE VARIABLES //
97 // ========================================================================== //
98 if (n > m+1) { std::cout << "error" << std::endl; return; }
99 
100 // Resize input variables
101 list.resize(n);
102 
103 // Initialize extraction set
104 N = m;
105 for (i = 0; i < m+1; i++) {
106  set[i] = i;
107 } //next i
108 
109 // ========================================================================== //
110 // EXTRACT INTEGERS WITHOUT REPLACEMENT //
111 // ========================================================================== //
112 for (i = 0; i < n; i++) {
113  index = (int) round(((double) N) * ((double) rand())/((double) RAND_MAX));
114  list[i] = set[index];
115  set[index] = set[N];
116  N--;
117 } //next i
