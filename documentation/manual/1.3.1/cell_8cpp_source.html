251 void Cell::setInterfaces(std::vector<std::vector<long>> &interfaces)
252 {
253  m_interfaces.initialize(interfaces);
254 
255  // The interface vector must have as many elements as faces
256  if (getType() != ElementInfo::UNDEFINED) {
257  int delta = m_interfaces.size() - getFaceCount();
258  if (delta > 0) {
259  for (int i = 0; i < delta; ++i) {
260  m_interfaces.popBack();
261  }
262  } else if (delta < 0) {
263  for (int i = 0; i < delta; ++i) {
264  m_interfaces.pushBack(1, NULL_ID);
265  }
266  }
267  }
268 
269  // Check that there is at least one interfaces for each face
270  for (int i = 0; i < m_interfaces.size(); ++i) {
271  if (m_interfaces.getItemCount(i) == 0) {
272  m_interfaces.pushBackItem(i, NULL_ID);
273  }
274  }

295 void Cell::pushInterface(const int &face, const long &interface)
296 {
297  // Do not push an existing interface
298  if (findInterface(face, interface) >= 0) {
299  return;
300  }
301 
302  // If there is only one interface stored for the specified face and
303  // that interface is negative, we need to overwrite the value and
304  // not add another interface.
305  if (m_interfaces.getItemCount(face) == 1) {
306  if (m_interfaces.getItem(face, 0) < 0) {
307  m_interfaces.setItem(face, 0, interface);
308  return;
309  }
310  }
311 
312  // There are multiple adjacency, we need to add the interface.
313  m_interfaces.pushBackItem(face, interface);
314 }

756 void Cell::display(std::ostream &out, unsigned short int indent) const
757 {
758  // ====================================================================== //
759  // VARIABLES DECLARATION //
760  // ====================================================================== //
761 
762  // Local variables
763  std::string t_s(indent, ' ');
764 
765  // Counters
766  int i, j;
767  int nn;
768 
769  // ====================================================================== //
770  // DISPLAY INFOS //
771  // ====================================================================== //
772  if (getType() == ElementInfo::UNDEFINED) {
773  out << t_s << "cell type: (unknown)" << std::endl;
774  return;
775  }
776 
777  // Scope variables -------------------------------------------------- //
778  int nv = getVertexCount();
779  int nf = getFaceCount();
780 
781  // General info ----------------------------------------------------- //
782  out << t_s << "cell type: " << getType() << std::endl;
783  out << t_s << "ID: " << getId() << std::endl;
784  out << t_s << "is ghost: ";
785  if (m_interior) { out << "(false)"; }
786  else { out << "(true)"; }
787  out << std::endl;
788 
789  // Connectivity infos --------------------------------------------------- //
790  out << t_s << "connectivity: [ ";
791  for (i = 0; i < nv-1; ++i) {
792  if (getVertex(i) == Vertex::NULL_ID) out << "n.a. ";
793  else out << getVertex(i) << ", ";
794  } //next i
795  if (getVertex(nv-1) == Vertex::NULL_ID) out << "n.a. ";
796  else out << getVertex(nv-1) << " ]" << std::endl;
797 
798  // neighbors infos ------------------------------------------------------ //
799  if (m_adjacencies.size() > 0) {
800  out << t_s << "neighbors: [ ";
801  for (i = 0; i < nf-1; ++i) {
802  nn = getAdjacencyCount(i);
803  out << "[ ";
804  for (j = 0; j < nn-1; ++j) {
805  if (getAdjacency(i,j) == Element::NULL_ID) out << "n.a. ";
806  else out << getAdjacency(i, j) << ", ";
807  } //next j
808  if (getAdjacency(i, nn-1) == Element::NULL_ID) out << "n.a. ], ";
809  else out << getAdjacency(i, nn-1) << " ], ";
810  } //next i
811  nn = getAdjacencyCount(nf-1);
812  out << "[ ";
813  for (j = 0; j < nn-1; ++j) {
814  if (getAdjacency(nf-1, j) == Element::NULL_ID) out << "n.a. ";
815  else out << getAdjacency(nf-1, j) << ", ";
816  } //next j
817  if (getAdjacency(nf-1, nn-1) == Element::NULL_ID) out << "n.a. ]";
818  else out << getAdjacency(nf-1, nn-1) << " ]";
819  out << " ]" << std::endl;
820  }
821 
822  // interface infos ------------------------------------------------------ //
823  if (m_interfaces.size() > 0) {
824  out << t_s << "interfaces: [ ";
825  for (i = 0; i < nf-1; ++i) {
826  nn = getInterfaceCount(i);
827  out << "[ ";
828  for (j = 0; j < nn-1; ++j) {
829  if (getInterface(i,j) == Element::NULL_ID) out << "n.a. ";
830  else out << getInterface(i, j) << ", ";
831  } //next j
832  if (getInterface(i, nn-1) == Element::NULL_ID) out << "n.a. ], ";

835  nn = getInterfaceCount(nf-1);
