277 void DataCommunicator::discoverSends()
278 {
279  // Cancel current sends
280  clearAllSends();
281 
282  // Send the data sizes with a synchronous send
283  int nRecvs = getRecvCount();
284 
285  std::vector<long> discoverSizes(nRecvs, 0);
286  std::vector<MPI_Request> discoverRequests(nRecvs, MPI_REQUEST_NULL);
287  for (int i = 0; i < nRecvs; ++i) {
288  int rank = m_recvRanks[i];
289  RecvBuffer &buffer = m_recvBuffers[i];
290  discoverSizes[i] = buffer.getSize();
291  MPI_Issend(discoverSizes.data() + i, 1, MPI_LONG, rank, m_discoverTag, m_communicator, discoverRequests.data() + i);
292  }
293 
294  // Receive the data sizes and set the sends
295  MPI_Request exchangeCompletedRequest = MPI_REQUEST_NULL;
296  while (true) {
297  // If there are messagea available receive them and set the sends
298  int messageAvailable = 1;
299  while (messageAvailable) {
300  MPI_Status status;
301  MPI_Iprobe(MPI_ANY_SOURCE, m_discoverTag, m_communicator, &messageAvailable, &status);
302  if (messageAvailable) {
303  long dataSize;
304  MPI_Recv(&dataSize, 1, MPI_LONG, status.MPI_SOURCE, m_discoverTag, m_communicator, MPI_STATUS_IGNORE);
305  setSend(status.MPI_SOURCE, dataSize);
306  }
307  }
308 
309  // If all the sends are complete notify it
310  if (exchangeCompletedRequest == MPI_REQUEST_NULL) {
311  int discoverSendsCompleted;
312  MPI_Testall(discoverRequests.size(), discoverRequests.data(), &discoverSendsCompleted, MPI_STATUSES_IGNORE);
313  if (discoverSendsCompleted) {
314  MPI_Ibarrier(m_communicator, &exchangeCompletedRequest);
315  }
316  }
317 
318  // If all sends are completed, check if also the other processes have
319  // completed the sends. Sice these are synchronous sends, they will
320  // be makred as completed only when the corresponding receive has
321  // completed. When all processes have completed the send/recevies
322  // all sizes have been exchanged.
323  if (exchangeCompletedRequest != MPI_REQUEST_NULL) {
324  int exchangeCompleted = 0;
325  MPI_Test(&exchangeCompletedRequest, &exchangeCompleted, MPI_STATUS_IGNORE);
326  if (exchangeCompleted) {
327  break;
328  }
329  }
330  }
331 
332  // At this point we are sure that all the processes have sent the data
333  // sizes and the data sizes have reached the destination, however it is
334  // not guaranteed that the data sizes have already been received (the
335  // message is on the destination process, but we don't know if it has
336  // been processed and received). Consequently, we can't be sure that all
337  // sends have been set.
338  //
339  // For the above reason it is not possible to call this function multiple
340  // times without receiving the messages or changes the exchanges between
341  // one call an another. Nor it is possible to call a discover recives
342  // followed by a discover sends (or viceversa) without receviing the
343  // messages or canceling them in between the two calls.
344 }

423 void DataCommunicator::clearSend(int rank)
424 {
425  if (m_sendIds.count(rank) == 0) {
426  return;
427  }
428 
429  // Cancel waiting send
430  cancelSend(rank);
431 
432  // Remove the send associated to the specified rank
433  int id = m_sendIds[rank];
434  m_sendIds.erase(rank);
435  for (auto &entry : m_sendIds) {
436  if (entry.second > id) {
437  entry.second--;
438  }
439  }
440 
441  m_sendRanks.erase(m_sendRanks.begin() + id);
442  m_sendRequests.erase(m_sendRequests.begin() + id);
443  m_sendBuffers.erase(m_sendBuffers.begin() + id);
444 }

714 void DataCommunicator::_startSend(int dstRank)
715 {
716  // Get the buffer
717  int id = m_sendIds.at(dstRank);
718  SendBuffer &sendBuffer = m_sendBuffers[id];
719  OBinaryStream &buffer = sendBuffer.getBack();
720 
721  // Start the send
722  int chunkSize = buffer.getChunkSize();
723  MPI_Datatype chunkDataType = getChunkDataType(chunkSize);
724 
725  MPI_Isend(buffer.data(), buffer.getChunkCount(), chunkDataType, dstRank, m_exchangeTag,
726  m_communicator, &m_sendRequests[id]);
727 }

783 int DataCommunicator::waitAnySend(const std::vector<int> &blackList)
784 {
785  // Exclude blackListed ranks
786  std::vector<MPI_Request> requestList(m_sendRequests);
787  for (const int rank : blackList) {
788  int id = m_sendIds.at(rank);
789  requestList[id] = MPI_REQUEST_NULL;
790  }
791 
792  // Wait for a send to complete
793  int id;
794  MPI_Waitany(requestList.size(), requestList.data(), &id, MPI_STATUS_IGNORE);
795  if (id == MPI_UNDEFINED) {
796  return MPI_UNDEFINED;
797  }
798 
799  m_sendRequests[id] = requestList[id];
800 
801  // Reset the position of the buffer
802  m_sendBuffers[id].seekg(0);
803 
804  // Return the rank associated to the completed send
805  return m_sendRanks[id];
806 }

1028 void DataCommunicator::cancelAllSends(bool synchronous)
1029 {
1030  // Cancel all the sends
1031  for (int rank : m_sendRanks) {
1032  cancelSend(rank);
1033  }
1034 
1035  // Early return if no synchronization is needed
1036  if (!synchronous) {
1037  return;
1038  }
1039 
1040  // Notify that the sends have been canceled
1041  //
1042  // The tag of the notifications has to be different from the tag used for
1043  // data size discovery. We may have a scenario like the following:
1044  //
1045  // discoverRecvs(); # <-- This call is using the discover tag
1046  //
1047  // startAllRecvs();

1058  //
1059  // cancelAllrecvs(true); # <-- This call is using the notification tag
1060  //
1061  // If a process completes all its exchanges, it may reach the function
1062  // that cancels the sends while other processes are still discovering
1063  // the receives. Thats's because data size discovery is a non-blocking
1064  // operation. At the end of data size discovery we know that the data
1065  // sizes for the recevies has reached the destination processes, but it
1066  // not guaranteed that the messages have been processed. What can happend
1067  // is that a process can exit from the function while other processes are
1068  // still receving/processing the recevied messages (so they are still
1069  // setting the receives). The notification issued when canceling the
1070  // sends may than be catched by the processes that are still discovering
1071  // the receives. To avoid this discover tags and notification tags have
1072  // to be different.
1073 
1074  // Start auxiliary receives for handling synchronization
1075  //
1076  // We will receive a notification from all the processes for which a
1077  // receive has been set. A notification equal to one means that the
1078  // send request on that processor that matches the receive of this rank has
1079  // been successfully cancelled.
1080  int nRecvNotifications = getRecvCount();
1081  std::vector<MPI_Request> recvNotificationRequests(nRecvNotifications);
1082  std::vector<int> remoteCancelCompleted(nRecvNotifications, 0);
1083  for (int i = 0; i < nRecvNotifications; ++i) {
1084  int rank = m_recvRanks[i];
1085  MPI_Irecv(remoteCancelCompleted.data() + i, 1, MPI_INT, rank, m_notificationTag, m_communicator, &recvNotificationRequests[i]);
1086  }
1087 
1088  // Notify that the sends have been canceled
1089  //
1090  // We will send a notification to all processes for wich a send has been
1091  // set. A notification equal to one means that the send request was
1092  // successfully cancelled.
1093  int nSendNotifications = getSendCount();
1094  std::vector<MPI_Request> sendNotificationRequests(nSendNotifications);
1095  std::vector<int> cancelCompleted(nSendNotifications, 1);
1096  for (int i = 0; i < nSendNotifications; ++i) {
1097  int rank = m_sendRanks[i];
1098  MPI_Isend(cancelCompleted.data() + i, 1, MPI_INT, rank, m_notificationTag, m_communicator, &sendNotificationRequests[i]);
1099  }
1100 
1101  // Synchronize with other processes
1102  //
1103  // Wait until all processes involved in data exchanging cancel their
1104  // send requests.
1105  MPI_Waitall(nRecvNotifications, recvNotificationRequests.data(), MPI_STATUS_IGNORE);
1106  for (int i = 0; i < nRecvNotifications; ++i) {
1107  if (!remoteCancelCompleted[i]) {
1108  log::cout() << "Unable to properly cancel the sends.";
1109  MPI_Abort(m_communicator, 1);
1110  }
1111  }
1112 
1113  // Wait until the notifications have been sent
1114  MPI_Waitall(nSendNotifications, sendNotificationRequests.data(), MPI_STATUS_IGNORE);
1115 
1116 #if BITPIT_ENABLE_DEBUG
1117  // Wait until all processes cancel the sends
1118  MPI_Barrier(m_communicator);
1119 #endif
1120 }
