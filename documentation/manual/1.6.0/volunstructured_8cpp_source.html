172 std::array<double, 3> VolUnstructured::evalInterfaceNormal(long id) const
173 {
174  std::array<std::array<double, 3>, ReferenceElementInfo::MAX_ELEM_VERTICES> coordinatesStaticPool;
175 
176  const Interface &interface = getInterface(id);
177  ConstProxyVector<std::array<double, 3>> vertexCoordinates = getElementVertexCoordinates(interface, coordinatesStaticPool.data());
178 
179  std::array<double, 3> orientation = {{0., 0., 0.}};
180  if (!isThreeDimensional()) {
181  long ownerId = interface.getOwner();
182  const Cell &owner = getCell(ownerId);
183 
184  ConstProxyVector<long> ownerVertexIds = owner.getVertexIds();
185  const int nOwnerVertices = ownerVertexIds.size();
186 
187  const std::array<double, 3> &V_A = getVertex(ownerVertexIds[0]).getCoords();
188  const std::array<double, 3> &V_B = getVertex(ownerVertexIds[1]).getCoords();
189  const std::array<double, 3> &V_Z = getVertex(ownerVertexIds[nOwnerVertices - 1]).getCoords();
190 
191  orientation = crossProduct(V_B - V_A, V_Z - V_A);
192  }
193 
194  return interface.evalNormal(vertexCoordinates.data(), orientation);
195 }
