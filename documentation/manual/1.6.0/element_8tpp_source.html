49 template<class DerivedElement>
50 ElementHalfItem<DerivedElement>::ElementHalfItem(DerivedElement &element, ConstProxyVector<long> &&connectivity, Winding winding)
51  : m_element(element), m_connect(std::move(connectivity)), m_winding(winding)
52 {
53  // Find the vertex with the lowest id, this will be used as first vertex
54  // when iterating over the connectivity.
55  std::size_t connectSize = m_connect.size();
56 
57  m_connectBegin = 0;
58  if (connectSize > 0) {
59  long smallestVertexId = m_connect[m_connectBegin];
60  for (std::size_t i = 1; i < connectSize; ++i) {
61  long vertexId = m_connect[i];
62  if (vertexId < smallestVertexId) {
63  m_connectBegin = i;
64  smallestVertexId = vertexId;
65  }
66  }
67  }
68 }

121 template<class DerivedElement>
122 bool ElementHalfItem<DerivedElement>::operator==(const ElementHalfItem<DerivedElement> &other) const
123 {
124  const bitpit::ConstProxyVector<long> &connect1 = m_connect;
125  const bitpit::ConstProxyVector<long> &connect2 = other.m_connect;
126 
127  std::size_t connectSize = connect1.size();
128  if (connectSize != connect2.size()) {
129  return false;
130  }
131 
132  int winding = m_winding * other.m_winding;
133  std::size_t offset = other.m_connectBegin - winding * m_connectBegin + 2 * connectSize;
134  for (std::size_t i = 0; i < connectSize; ++i) {
135  std::size_t k1 = i;
136  std::size_t k2 = (offset + winding * i) % connectSize;
137  if (connect1[k1] != connect2[k2]) {
138  return false;
139  }
140  }
141 
142  return true;
143 }

172 template<class DerivedElement>
173 std::size_t ElementHalfItem<DerivedElement>::Hasher::operator()(const ElementHalfItem &item) const
174 {
175  const ConstProxyVector<long> &connectivity = item.m_connect;
176  std::size_t connectivitySize = connectivity.size();
177 
178  std::size_t hash = connectivitySize;
179  if (item.m_winding == ElementHalfItem::WINDING_NATURAL) {
180  for (std::size_t i = 0; i < connectivitySize; ++i) {
181  std::size_t k = (item.m_connectBegin + i) % connectivitySize;
182  utils::hashing::hash_combine(hash, connectivity[k]);
183  }
184  } else {
185  for (std::size_t i = connectivitySize; i > 0; --i) {
186  std::size_t k = (item.m_connectBegin + i) % connectivitySize;
187  utils::hashing::hash_combine(hash, connectivity[k]);
188  }
189  }
190 
191  return hash;
192 }
