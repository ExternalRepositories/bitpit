342 std::vector<adaption::Info> PatchKernel::update(bool trackAdaption, bool squeezeStorage)
343 {
344  std::vector<adaption::Info> updateInfo;
345 
346  // Check if there are pending changes
347  bool spawnNeeed = (getSpawnStatus() == SPAWN_NEEDED);
348  bool adaptionDirty = (getAdaptionStatus(true) == ADAPTION_DIRTY);
349  bool boundingBoxDirty = isBoundingBoxDirty();
350 
351  bool pendingChanges = (spawnNeeed || adaptionDirty || boundingBoxDirty);
352  if (!pendingChanges) {
353  return updateInfo;
354  }
355 
356  // Spawn
357  if (spawnNeeed) {
358  mergeAdaptionInfo(spawn(trackAdaption), updateInfo);
359  }
360 
361  // Adaption
362  if (adaptionDirty) {
363  mergeAdaptionInfo(adaption(trackAdaption, squeezeStorage), updateInfo);
364  }
365 
366  // Update bounding box
367  updateBoundingBox();
368 
369  return updateInfo;
370 }

399 std::vector<adaption::Info> PatchKernel::spawn(bool trackSpawn)
400 {
401  std::vector<adaption::Info> spawnInfo;
402 
403 #if BITPIT_ENABLE_MPI==1
404  // This is a collevtive operation and should be called by all processes
405  if (isCommunicatorSet()) {
406  const auto &communicator = getCommunicator();
407  MPI_Barrier(communicator);
408  }
409 #endif
410 
411  // Check spawn status
412  SpawnStatus spawnStatus = getSpawnStatus();
413  if (spawnStatus == SPAWN_UNNEEDED || spawnStatus == SPAWN_DONE) {
414  return spawnInfo;
415  }
416 
417  // Begin patch alteration
418  beginAlteration();
419 
420  // Alter patch
421  spawnInfo = _spawn(trackSpawn);
422 
423  // End patch alteration
424  endAlteration(true);
425 
426  // Spwan is done
427  setSpawnStatus(SPAWN_DONE);
428 
429  // Done
430  return spawnInfo;

586 {
587  // Flush data structures
588  m_cells.flush();
589  m_interfaces.flush();
590  m_vertices.flush();
591 
592  // Squeeze data structures
593  if (squeezeStorage) {
594  squeeze();
595  }
596 
597  // Update geometric information
598  updateBoundingBox();
599 
600 #if BITPIT_ENABLE_MPI==1
601  // Update information for ghost data exchange
602  //
603  // If we are partitioning the patch, the partition flag is not set yet (it
604  // will be set after the call to this function). We call the update of
605  // information for ghost data exchange unconditionally, if there are no
606  // ghosts the function will exit without doing anything.
607  if (getProcessorCount() > 1 && getAdjacenciesBuildStrategy() != ADJACENCIES_NONE) {
608  updateGhostExchangeInfo();
609  }
610 #endif
611 
612  // Synchronize storage
613  m_cells.sync();
614  m_interfaces.sync();
615  m_vertices.sync();
616 }

857 void PatchKernel::write(VTKWriteMode mode)
858 {
859  // Get VTK cell count
860  long vtkCellCount = 0;
861  if (m_vtkWriteTarget == WRITE_TARGET_CELLS_ALL) {
862  vtkCellCount = getCellCount();
863 #if BITPIT_ENABLE_MPI==1
864  } else if (m_vtkWriteTarget == WRITE_TARGET_CELLS_INTERNAL) {
865  vtkCellCount = getInternalCount();
866 #endif
867  }
868 
869  // Set the dimensinos of the mesh
870  PiercedStorage<bool, long> vertexWriteFlag(1, &m_vertices);
871  vertexWriteFlag.fill(false);
872 
873  bool vtkFaceStreamNeeded = false;
874  for (const Cell &cell : m_cells) {
875  if (cell.getDimension() > 2 && !cell.hasInfo()) {
876  vtkFaceStreamNeeded = true;
877  break;
878  }
879  }
880 
881  long vtkConnectSize = 0;
882  long vtkFaceStreamSize = 0;
883  for (const Cell &cell : getVTKCellWriteRange()) {
884  ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
885  const int nCellVertices = cellVertexIds.size();
886  for (int k = 0; k < nCellVertices; ++k) {
887  long vertexId = cellVertexIds[k];
888  vertexWriteFlag.at(vertexId) = true;
889  }
890 
891  vtkConnectSize += nCellVertices;
892  if (vtkFaceStreamNeeded) {
893  if (cell.getDimension() <= 2 || cell.hasInfo()) {
894  vtkFaceStreamSize += 1;
895  } else {
896  vtkFaceStreamSize += cell.getFaceStreamSize();
897  }
898  }
899  }
900 
901  int vtkVertexCount = 0;
902  m_vtkVertexMap.unsetKernel(true);
903  m_vtkVertexMap.setStaticKernel(&m_vertices);
904  for (VertexConstIterator itr = vertexConstBegin(); itr != vertexConstEnd(); ++itr) {
905  std::size_t vertexRawId = itr.getRawIndex();
906  if (vertexWriteFlag.rawAt(vertexRawId)) {
907  m_vtkVertexMap.rawAt(vertexRawId) = vtkVertexCount++;
908  } else {
909  m_vtkVertexMap.rawAt(vertexRawId) = Vertex::NULL_ID;
910  }
911  }
912 
913  m_vtk.setDimensions(vtkCellCount, vtkVertexCount, vtkConnectSize, vtkFaceStreamSize);
914 
915  // Write the mesh
916  m_vtk.write(mode);
917 }

4349 void PatchKernel::updateAdjacencies(const std::vector<long> &cellIds)
4350 {
4351  // The adjacencies are found looking for matching half-faces.
4352  //
4353  // On a three-dimensional patch each internal face is shared between two,
4354  // and only two, half-faces. On lower dimension patches one internal face
4355  // may be shared among multiple half-faces (this happend with non-manifold
4356  // patches).
4357  //
4358  // If faces can be shared only between two half-faces, there is a one-to-one
4359  // correspondence between the half-faces. Give one half-faces, its matching
4360  // can be found looking for an half-faces with the same vertices but with
4361  // reverse vertex winding order.
4362  //
4363  // If faces can be shared among more than two half-faces, it's not anymore
4364  // possible to define a unique winding order for the vertices of the faces.
4365  // There is no more a one-to-one correspondence between pairs of half-face.
4366  // In this case, when looking for half-face correspondence, it is necessary
4367  // to look for half-faces with reverse vertex winding order and also for
4368  // half-faces with the same vertex winding order.
4369  //
4370  // If we are updating the adjacencies of only part of the cells, we need
4371  // to populate the half-face list with the faces of the cells that will
4372  // not be updated (among those faces may find matches for the faces of
4373  // the updated cells). If faces can be shared only between two half-faces
4374  // we need to insert only the border faces of the non-updated cells,
4375  // otherwise we need to add all the faces of the non-updated cells.
4376 
4377  // Define matching windings
4378  bool multipleMatchesAllowed = (getDimension() < 3);
4379 
4380  std::vector<CellHalfFace::Winding> matchingWindings;
4381  matchingWindings.push_back(CellHalfFace::WINDING_REVERSE);
4382  if (multipleMatchesAllowed) {
4383  matchingWindings.push_back(CellHalfFace::WINDING_NATURAL);
4384  }
4385 
4386  // Initialize half-faces list
4387  std::unordered_multiset<CellHalfFace, CellHalfFace::Hasher> halfFaces;
4388  if (multipleMatchesAllowed) {
4389  halfFaces.reserve(4 * getCellCount());

4391  halfFaces.reserve(2 * cellIds.size());

4394  // Populate list with faces of non-updated cells
4395  if ((long) cellIds.size() != getCellCount()) {

4397 
4398  for (Cell &cell : m_cells) {
4399  if (updateCellSet.count(cell.getId()) > 0) {
4400  continue;
4401  }
4402 
4403  int nCellFaces = cell.getFaceCount();
4404  for (int face = 0; face < nCellFaces; face++) {
4405  if (multipleMatchesAllowed || cell.isFaceBorder(face)) {
4406  halfFaces.emplace(cell, face, CellHalfFace::WINDING_NATURAL);
4407  }
4408  }
4409  }
4410  }
4411 
4412  // Update the adjacencies
4413  for (long cellId : cellIds) {
4414  Cell &cell = m_cells[cellId];
4415 
4416  const int nCellFaces = cell.getFaceCount();
4417  for (int face = 0; face < nCellFaces; face++) {
4418  // Generate the half-face
4419  CellHalfFace halfFace(cell, face);
4420 
4421  // Find matching half-faces
4422  int nAdjacencies = 0;
4423  for (CellHalfFace::Winding winding : matchingWindings) {
4424  // Set winding order
4425  halfFace.setWinding(winding);
4426 
4427  // Search for matching half-faces
4428  //
4429  // Each match is adjacent to the current half-face.
4430  auto matchRange = halfFaces.equal_range(halfFace);
4431  for (auto matchItr = matchRange.first; matchItr != matchRange.second; ++matchItr) {
4432  const CellHalfFace &neighHalfFace = *matchItr;
4433 
4434  // Generate the adjacency
4435  Cell &neigh = neighHalfFace.getCell();
4436  long neighId = neigh.getId();
4437  long neighFace = neighHalfFace.getFace();
4438 
4439  cell.pushAdjacency(face, neighId);
4440  neigh.pushAdjacency(neighFace, cellId);
4441 
4442  ++nAdjacencies;
4443  }
4444 
4445  // Remove the matching half-faces from the list
4446  //
4447  // It is not possible to remove the matching half-faces
4448  // if multiple matches are allowed.
4449  if (!multipleMatchesAllowed) {
4450  halfFaces.erase(matchRange.first, matchRange.second);
4451  }
4452  }
4453

4455  //
4456  // If no multiple matches are allowed, we need to add the current
4457  // half-face to the list only if no matchings were found.
4458  if (multipleMatchesAllowed || nAdjacencies == 0) {
4459  halfFace.setWinding(CellHalfFace::WINDING_NATURAL);

4461  }
4462  }
4463  }
