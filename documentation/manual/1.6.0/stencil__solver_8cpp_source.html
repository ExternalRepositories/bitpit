159 void StencilScalarSolver::solve()
160 {
161  // Check if the stencil solver is assembled
162  if (!isAssembled()) {
163  throw std::runtime_error("Unable to solve the system. The stencil solver is not yet assembled.");
164  }
165 
166  // Subtract constant terms to the RHS
167  long nUnknowns = getRowCount();
168  double *raw_rhs = getRHSRawPtr();
169  for (long i = 0; i < nUnknowns; ++i) {
170  raw_rhs[i] -= m_constants[i];
171  }
172  restoreRHSRawPtr(raw_rhs);
173 
174  // Solve the system
175  SystemSolver::solve();
176 }
