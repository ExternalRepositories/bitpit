243 double LevelSetBoolean::getMinSurfaceFeatureSize() const {
244 
245  bool minimumValid = false;
246  double minimumSize = levelSetDefaults::SIZE;
247  for( const auto & object : m_objPtr ){
248  double objectMinimumSize = object->getMinSurfaceFeatureSize();
249  if (objectMinimumSize < 0) {
250  continue;
251  }
252 
253  minimumValid = true;
254  minimumSize = std::min(objectMinimumSize, minimumSize);
255  }
256 
257  if (!minimumValid) {
258  minimumSize = - levelSetDefaults::SIZE;
259  }
260 
261  return minimumSize;
262 }

279 /*
280  * Determines the relevant object which determines the levelset value in the cell
281  * Taken from http://www.iue.tuwien.ac.at/phd/ertl/node57.html
282  * @param[in] id cell index
283  * @return pointer to competent LevelSetObject
284  */
285 LevelSetObject* LevelSetBoolean::getCompetentObject( const long &id) const{
286 
287  if(m_objPtr.size()==0){
288  return nullptr;
289  }
290 
291  double result, second;
292  LevelSetObject *resPtr, *secPtr;
293 
294  result = m_objPtr[0]->getLS(id);
295  resPtr = m_objPtr[0];
296 
297  for( size_t n=1; n<m_objPtr.size(); ++n){
298  second = m_objPtr[n]->getLS(id) ;
299  secPtr = m_objPtr[n];
300 
301  if( getBooleanOperation() == LevelSetBooleanOperation::UNION){
302  if(result>second) {
303  result = second;
304  resPtr = secPtr;
305  }
306 
307  } else if ( getBooleanOperation() == LevelSetBooleanOperation::INTERSECTION){
308  if(result<second) {
309  result = second;
310  resPtr = secPtr;
311  }
312 
313  } else if ( getBooleanOperation() == LevelSetBooleanOperation::SUBTRACTION){
314  if(result<-1.*second) {
315  result = -1.*second;
316  resPtr = secPtr;
317  }
318  }
319  }
320 
321  return resPtr;
322 }

330 LevelSetInfo LevelSetBoolean::booleanOperation(const long &id) const{
331 
332  if(m_objPtr.size()==0){
333  return LevelSetInfo();
334  }
335 
336  LevelSetInfo result, second;
337  result = m_objPtr[0]->getLevelSetInfo(id);
338 
339  for( size_t n=1; n<m_objPtr.size(); ++n){
340  second = m_objPtr[n]->getLevelSetInfo(id) ;
341 
342  if( getBooleanOperation() == LevelSetBooleanOperation::UNION){
343  if(result.value>second.value) {
344  result = second;
345  }
346 
347  } else if ( getBooleanOperation() == LevelSetBooleanOperation::INTERSECTION){
348  if(result.value<second.value) {
349  result = second;
350  }
351 
352  } else if ( getBooleanOperation() == LevelSetBooleanOperation::SUBTRACTION){
353  if(result.value<-1.*second.value) {
354  second.value *= -1.;
355  second.gradient *= -1.;
356  result = second;
357  }
358  }
359  }
360 
361  return result;
362 }
