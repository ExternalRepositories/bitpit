166 void PatchGlobalInfo::_extract()
167 {
168  long globalId;
169 
170  size_t exchangeDataSize = sizeof(globalId);
171  std::unique_ptr<DataCommunicator> dataCommunicator;
172 
173  if (m_patch->getProcessorCount() > 1) {
174  // Create the data communicator
175  dataCommunicator = std::unique_ptr<DataCommunicator>(new DataCommunicator(m_patch->getCommunicator()));
176  dataCommunicator->setTag(108);
177 
178  // Set and start the receives
179  for (const auto entry : m_patch->getGhostExchangeTargets()) {
180  const int rank = entry.first;
181  const auto &list = entry.second;
182 
183  dataCommunicator->setRecv(rank, list.size() * exchangeDataSize);
184  dataCommunicator->startRecv(rank);
185  }
186  }
187 
188  // Get the internal count of all the partitions
189  m_nGlobalInternals.resize(m_patch->getProcessorCount());
190  if (m_patch->getProcessorCount() > 1) {
191  long nLocalInternals = m_patch->getInternalCount();
192  MPI_Allgather(&nLocalInternals, 1, MPI_LONG, m_nGlobalInternals.data(), 1, MPI_LONG, m_patch->getCommunicator());
193  } else {
194  m_nGlobalInternals[0] = m_patch->getInternalCount();
195  }
196 
197  // Evaluate the offset for the current partition
198  long offset = getCellGlobalOffset();
199 
200  // Evalaute the global id of the internal cells
201  if (m_patch->getInternalCount() > 0) {
202  std::map<long,long> nativeIds;
203  for (auto itr = m_patch->internalConstBegin(); itr != m_patch->internalConstEnd(); ++itr) {
204  long id = itr.getId();
205  long nativeId = m_patch->_getCellNativeIndex(id);
206 
207  nativeIds.insert({nativeId, id});
208  }
209 
210  globalId = offset;
211  for (auto itr = nativeIds.begin(); itr != nativeIds.end(); ++itr) {
212  m_cellLocalToGlobalMap.insert({itr->second, globalId++});
213  }
214 
215  std::map<long,long>().swap(nativeIds);
216  }
217 
218  // Communicate the global id of the ghost cells
219  if (m_patch->getProcessorCount() > 1) {
220  // Set and start the sends
221  for (const auto entry : m_patch->getGhostExchangeSources()) {
222  const int rank = entry.first;
223  auto &list = entry.second;
224 
225  dataCommunicator->setSend(rank, list.size() * exchangeDataSize);
226  SendBuffer &buffer = dataCommunicator->getSendBuffer(rank);
227  for (long id : list) {
228  buffer << m_cellLocalToGlobalMap.at(id);
229  }
230  dataCommunicator->startSend(rank);
231  }
232 
233  // Receive the global ids of the ghosts
234  int nCompletedRecvs = 0;
235  while (nCompletedRecvs < dataCommunicator->getRecvCount()) {
236  int rank = dataCommunicator->waitAnyRecv();
237  const auto &list = m_patch->getGhostExchangeTargets(rank);
238 
239  RecvBuffer &buffer = dataCommunicator->getRecvBuffer(rank);
240  for (long id : list) {
241  buffer >> globalId;
242  m_cellLocalToGlobalMap.insert({id, globalId});
243  }
244 
245  ++nCompletedRecvs;
246  }
247 
248  // Wait for the sends to finish
249  dataCommunicator->waitAllSends();
250  }
251 }

291 int PatchGlobalInfo::getCellRankFromGlobal(long id) const

295  offset += m_nGlobalInternals[k];
