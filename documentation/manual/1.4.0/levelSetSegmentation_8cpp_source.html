109 void SegmentationKernel::setSurface( const SurfUnstructured *surface, double featureAngle){
110 
111  std::vector<std::array<double,3>> vertexNormal ;
112  std::vector<std::array<double,3>> vertexGradient ;
113 
114  m_surface = surface;
115  m_featureAngle = featureAngle;
116 
117  double tol = m_surface->getTol() ;
118  for( const Cell &segment : m_surface->getCells() ){
119  long segmentId = segment.getId() ;
120  int nVertices = segment.getVertexCount() ;
121 
122  vertexNormal.resize(nVertices) ;
123  vertexGradient.resize(nVertices) ;
124 
125  double misalignment = 0. ;
126  for( int i = 0; i < nVertices; ++i ){
127  vertexGradient[i] = m_surface->evalVertexNormal(segmentId, i) ;
128  vertexNormal[i] = m_surface->evalLimitedVertexNormal(segmentId, i, m_featureAngle) ;
129 
130  misalignment += norm2(vertexGradient[i] - vertexNormal[i]) ;
131  }
132 
133  m_vertexGradients.insert({{segmentId, vertexGradient}}) ;
134  if( misalignment >= tol ){
135  m_vertexNormals.insert({{segmentId, vertexNormal}}) ;
136  }
137  }
138 
139  // Initialize search tree
140  m_searchTreeUPtr = std::unique_ptr<SurfaceSkdTree>(new SurfaceSkdTree(surface));
141  m_searchTreeUPtr->build();
142 }

176 void SegmentationKernel::getSegmentInfo( const std::array<double,3> &pointCoords, const long &segmentId, const bool &signd, double &distance, std::array<double,3> &gradient, std::array<double,3> &normal ) const {
177 
178 
179  std::array<double,3> outwards;
180 
181  auto itrNormal = getVertexNormals().find(segmentId) ;
182  auto itrGradient = getVertexGradients().find(segmentId) ;
183  assert( itrGradient != getVertexGradients().end() ) ;
184 
185  const Cell &cell = m_surface->getCell(segmentId) ;
186  ElementType cellType = cell.getType();
187  const long *cellConnect = cell.getConnect();
188 
189  std::array<double,3> projectionCoords;
190 
191  switch (cellType) {
192 
193  case ElementType::VERTEX :
194  {
195  long id = cellConnect[0] ;
196 
197  projectionCoords = m_surface->getVertexCoords(id);
198 
199 
200  normal.fill(0.);
201 
202  break;
203  }
204 
205  case ElementType::LINE:
206  {
207  long id0 = cellConnect[0] ;
208  long id1 = cellConnect[1] ;
209  std::array<double,2> lambda ;
210 
211  projectionCoords = CGElem::projectPointSegment( pointCoords, m_surface->getVertexCoords(id0), m_surface->getVertexCoords(id1), lambda);
212  outwards = lambda[0] *itrGradient->second[0] ;
213  outwards += lambda[1] *itrGradient->second[1] ;
214  outwards /= norm2(outwards) ;
215 
216  if( itrNormal != getVertexNormals().end() ){
217  normal = lambda[0] *itrNormal->second[0] ;
218  normal += lambda[1] *itrNormal->second[1] ;
219  normal /= norm2(normal) ;
220 
221  } else {
222  normal = outwards;
223 
224  }
225 
226  break;
227  }
228 
229  case ElementType::TRIANGLE:
230  {
231  long id0 = cellConnect[0] ;
232  long id1 = cellConnect[1] ;
233  long id2 = cellConnect[2] ;
234 
235  std::array<double,3> lambda ;
236 
237  projectionCoords = CGElem::projectPointTriangle( pointCoords, m_surface->getVertexCoords(id0), m_surface->getVertexCoords(id1), m_surface->getVertexCoords(id2), lambda );
238  outwards = lambda[0] *itrGradient->second[0] ;
239  outwards += lambda[1] *itrGradient->second[1] ;
240  outwards += lambda[2] *itrGradient->second[2] ;
241  outwards /= norm2(outwards);
242 
243 
244  if( itrNormal != getVertexNormals().end() ){
245  normal = lambda[0] *itrNormal->second[0] ;
246  normal += lambda[1] *itrNormal->second[1] ;
247  normal += lambda[2] *itrNormal->second[2] ;
248  normal /= norm2(normal) ;
249 
250  } else {
251  normal = outwards;
252 
253  }
254 
255  break;
256  }
257 
258  default:
259  {
260  std::runtime_error ("Type of cell not supported.");
261  break;
262  }
263 
264  }
265 
266  gradient = pointCoords-projectionCoords;
267  distance = norm2(gradient);
268  gradient /= distance;
269 
270  // the sign is computed by determining the side of point p
271  // with respect to the normal plane
272  double s = sign( dotProduct(gradient, outwards) );
273 
274  // if p lies on the normal plane (s=0), but the distance is finite the sign must be evaluated
275  // considering the curvature of the surface. Anyhow this situation is not crucial because
276  // there should exists another element with smaller distance. The signed is put aribtrarly
277  // to positive
278  if(utils::DoubleFloatingEqual()(s,0.) && distance>0){
279  s = 1.;
280  }
281 
282  // If signed distance are computed, the distance value and gradient
283  // need to be changed accordingly. If unsigned distance are computed
284  // the orientation of the suraface normal is discarded and in order
285  // to agnostic with repect the two sides of the surface
286  distance *= ( signd *s + (!signd) *1.);
287  gradient *= ( signd *s + (!signd) *1.);
288  normal *= ( signd *1. + (!signd) *s );
289 
290  return;
291 
292 }

524 bool LevelSetSegmentation::seedNarrowBand( LevelSetCartesian *visitee, std::vector<std::array<double,3>> &VS, double searchRadius, std::vector<long> &I){
525 
526  VolCartesian &mesh = *(static_cast<VolCartesian*>(visitee->getMesh()));
527 
528  bool found(false) ;
529  int dim( mesh.getDimension() ) ;
530  std::array<double,3> B0, B1;
531  std::vector<std::array<double,3>> VP ;
532 
533  mesh.getBoundingBox(B0, B1) ;
534 
535  for( int i=0; i<dim; ++i){
536  B0[i] -= searchRadius;
537  B1[i] += searchRadius;
538  }
539 
540  I.clear() ;
541 
542  for( const auto &P : VS){
543  if( CGElem::intersectPointBox( P, B0, B1, dim ) ) {
544  I.push_back( mesh.locateClosestCell(P) );
545  found = true ;
546  }
547  }
548 
549  if( !found && CGElem::intersectBoxPolygon( B0, B1, VS, false, true, true, VP, dim ) ) {
550  for( const auto &P : VP){
551  I.push_back( mesh.locateClosestCell(P) );
552  found = true ;
553  }
554  }
555 
556  return found ;
557 }

620 void LevelSetSegmentation::computeLSInNarrowBand( LevelSetCartesian *visitee, bool signd){
621 
622  VolCartesian &mesh = *(visitee->getCartesianMesh() ) ;
623  double searchRadius = m_narrowBand;
624 
625  if(searchRadius<0.){
626  for( int d=0; d < mesh.getDimension(); ++d){
627  searchRadius = std::max( searchRadius, mesh.getSpacing(d) ) ;
628  }
629  }
630 
631  std::vector<std::array<double,3>> VS(3);
632 
633  const SurfUnstructured &m_surface = m_segmentation->getSurface();
634 
635  std::vector<long> stack, temp, neighs, flag( mesh.getCellCount(), -1);
636 
637  std::vector< std::array<double,3> > cloud ;
638  std::vector<double> cloudDistance;
639 
640  double distance;
641  std::array<double,3> gradient, normal;
642 
643  stack.reserve(128) ;
644  temp.reserve(128) ;
645 
646  log::cout() << " Compute levelset on cartesian mesh" << std::endl;
647 
648  for (const Cell &segment : m_surface.getCells()) {
649  long segmentId = segment.getId();
650 
651  // compute initial seeds, ie the cells where the vertices
652  // of the surface element fall in and add them to stack
653  m_segmentation->getSegmentVertexCoords( segmentId, &VS ) ;
654  seedNarrowBand( visitee, VS, searchRadius, stack );
655 
656  ElementType segmentType = segment.getType();
657 
658 
659  // propagate from seed
660  size_t stackSize = stack.size();
661  while (stackSize > 0) {
662 
663  // put the cell centroids of the stack into a vector
664  // and calculate the distances to the cloud
665  cloud.resize(stackSize) ;
666  cloudDistance.resize(stackSize);
667 
668  for( size_t k = 0; k < stackSize; ++k) {
669  long cell = stack[k];
670  cloud[k] = visitee->computeCellCentroid(cell) ;
671  }
672 
673  switch (segmentType) {
674 
675  case ElementType::VERTEX :
676  {
677  for( size_t k=0; k<stackSize; ++k){
678  cloudDistance[k] = norm2( cloud[k]-VS[0]);
679  }
680  break;
681  }
682 
683  case ElementType::LINE:
684  {
685  for( size_t k=0; k<stackSize; ++k){
686  cloudDistance[k] = CGElem::distancePointSegment( cloud[k], VS[0], VS[1]);
687  }
688  break;
689  }
690 
691  case ElementType::TRIANGLE:
692  {
693  cloudDistance = CGElem::distanceCloudTriangle( cloud, VS[0], VS[1], VS[2]);
694  break;
695  }
696 
697  default:
698  {
699  std::runtime_error ("Type of cell not supported.");
700  break;
701  }
702  }
703 
704  // check each cell of cloud individually
705  for( size_t k = 0; k < stackSize; ++k) {
706 
707  long &cellId = stack[k];
708  double &cellDistance = cloudDistance[k];
709 
710  // consider only cells within the search radius
711  if ( cellDistance <= searchRadius ) {
712 
713  PiercedVector<LevelSetInfo>::iterator lsInfoItr = m_ls.find(cellId) ;
714  if( lsInfoItr == m_ls.end() ){
715  lsInfoItr = m_ls.emplace(cellId) ;
716  }
717 
718  // check if the computed distance is the closest distance
719  if( cellDistance < std::abs(lsInfoItr->value) ){
720 
721  // compute all necessary information and store them
722  m_segmentation->getSegmentInfo(cloud[k], segmentId, signd, distance, gradient, normal);
723 
724  lsInfoItr->value = distance;
725  lsInfoItr->gradient = gradient;
726 
727  auto infoItr = m_surfaceInfo.find(cellId) ;
728  if( infoItr == m_surfaceInfo.end() ){
729  infoItr = m_surfaceInfo.emplace(cellId) ;
730  }
731  infoItr->support = segmentId;
732  infoItr->normal = normal;
733  }
734 
735 
736  // the new stack is composed of all neighbours
737  // of the old stack. Attention must be paid in
738  // order not to evaluate the same cell twice
739  neighs.clear();
740  mesh.findCellFaceNeighs(cellId, &neighs) ;
741  for( const auto & neigh : neighs){
742  if( flag[neigh] != segmentId) {
743  temp.push_back( neigh) ;
744  flag[neigh] = segmentId ;
745  }
746  }
747 
748  } //end if distance
749  }
750 
751  stack.clear() ;
752  stack.swap( temp ) ;
753  stackSize = stack.size() ;
754 
755  } //end while
756  }
757 
758  return;
759 }

763 void LevelSetSegmentation::computeLSInNarrowBand( LevelSetOctree *visitee, bool signd){
764 
765  VolumeKernel &mesh = *(visitee->getMesh()) ;
766 
767  bool adaptiveSearch(m_narrowBand<0);
768  double searchRadius = m_narrowBand;
769  double factor = 0.5 *sqrt( (double) mesh.getDimension() );
770 
771  long segmentId;
772  double distance;
773  std::array<double,3> gradient, normal;
774  std::array<double,3> centroid, root;
775 
776  std::unordered_set<long> intersects, prcoessed;
777 
778  for( const Cell &cell : mesh.getCells() ){
779 
780  long cellId = cell.getId();
781 
782  centroid = visitee->computeCellCentroid(cellId);
783 
784  if(adaptiveSearch){
785  double cellSize = mesh.evalCellSize(cellId);
786  searchRadius = factor *cellSize;
787  }
788 
789  m_segmentation->m_searchTreeUPtr->findPointClosestCell(centroid, searchRadius, &segmentId, &distance);
790 
791  if(segmentId>=0){
792 
793  m_segmentation->getSegmentInfo(centroid, segmentId, signd, distance, gradient, normal);
794 
795  PiercedVector<LevelSetInfo>::iterator lsInfoItr = m_ls.find(cellId);
796  lsInfoItr = m_ls.emplace(cellId) ;
797  lsInfoItr->value = distance;
798  lsInfoItr->gradient = gradient;
799 
800  PiercedVector<SurfaceInfo>::iterator infoItr = m_surfaceInfo.emplace(cellId);
801  infoItr->support = segmentId;
802  infoItr->normal = normal;
803 
804  if(adaptiveSearch){
805  intersects.insert(cellId);
806  }
807 
808  }
809 
810  }
811 
812  if(!adaptiveSearch){
813  return;
814  }
815 
816  for( const long &cellId : intersects){
817 
818  Cell const &cell = mesh.getCell(cellId);
819 
820  centroid = visitee->computeCellCentroid(cellId);
821  root = computeProjectionPoint(cellId);
822 
823  const long *neighbours = cell.getAdjacencies() ;
824  int nNeighbours = cell.getAdjacencyCount() ;
825  for (int n = 0; n < nNeighbours; ++n) {
826  long neighId = neighbours[n];
827 
828  if (neighId < 0) {
829  continue;
830  }
831 
832  // skip if neigh cell has already been processed
833  // either because it is intersected by surface or
834  // because it is a neigh to a previous intersect
835  if( m_ls.count(neighId) != 0 ){
836  continue;
837  }
838 
839  centroid = visitee->computeCellCentroid(neighId);
840  searchRadius = 1.05 *norm2(centroid-root);
841  m_segmentation->m_searchTreeUPtr->findPointClosestCell(centroid, searchRadius, &segmentId, &distance);
842 
843  if(segmentId>=0){
844 
845  m_segmentation->getSegmentInfo(centroid, segmentId, signd, distance, gradient, normal);
846 
847  PiercedVector<LevelSetInfo>::iterator lsInfoItr = m_ls.emplace(neighId) ;
848  lsInfoItr->value = distance;
849  lsInfoItr->gradient = gradient;
850 
851 
852  PiercedVector<SurfaceInfo>::iterator infoItr = m_surfaceInfo.emplace(neighId);
853  infoItr->support = segmentId;
854  infoItr->normal = normal;
855 
856  } else {
857  assert(false && "Should not pass here");
858  }
859 
860  }
861  }
862 }

866 void LevelSetSegmentation::updateLSInNarrowBand( LevelSetOctree *visitee, const std::vector<adaption::Info> &mapper, bool signd){
867 
868  VolumeKernel &mesh = *(visitee->getMesh()) ;
869
