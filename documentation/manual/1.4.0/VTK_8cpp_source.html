603 void VTK::calcAppendedOffsets(){
604 
605  uint64_t offset(0) ;
606  uint64_t HeaderByte(0) ;
607 
608  if( getHeaderType() == "UInt32"){
609  HeaderByte = sizeof(uint32_t) ;
610  }
611 
612  else if( getHeaderType() == "UInt64") {
613  HeaderByte = sizeof(uint64_t) ;
614  }
615 
616  for( auto & field : m_data ){
617  if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED && field.getLocation() == VTKLocation::POINT ) {
618  field.setOffset( offset) ;
619  offset += HeaderByte + calcFieldSize(field) ;
620  }
621  }
622 
623  for( auto & field : m_data ){
624  if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED && field.getLocation() == VTKLocation::CELL) {
625  field.setOffset( offset) ;
626  offset += HeaderByte + calcFieldSize(field) ;
627  }
628  }
629 
630  for( auto & field : m_geometry ){
631  if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED ) {
632  field.setOffset( offset) ;
633  offset += HeaderByte + calcFieldSize(field) ;
634  }
635  }

724  std::fstream str ;
725  std::fstream::pos_type position_insert, position_eof ;
726 
727  int length;
728  char* buffer ;
729 
730  str.open( m_fh.getPath( ), std::ios::in | std::ios::out ) ;
731 
732  { // Write Ascii
733 
734  position_insert = str.tellg();
735 
736  //Writing first point data then cell data
737  for( auto &field : m_data ){
738  if( field.isEnabled() && field.getCodification() == VTKFormat::ASCII && field.getLocation() == VTKLocation::POINT ) {
739  str.seekg( position_insert);
740  readDataArray( str, field ) ;
741 
742  str.seekg( field.getPosition() ) ;
743  genericIO::copyUntilEOFInString( str, buffer, length );
744 
745  field.write( str ) ;
746  position_insert = str.tellg();
747 
748  if(position_insert==field.getPosition()){
749  log::cout() << "Error VTK: No data has been written for field " << field.getName() << std::endl;
750  assert(false);
751  }
752 
753  str << std::endl ;
754  genericIO::flushBINARY( str, buffer, length) ;
755 
756  delete [] buffer ;
757 
758  }
759  }
760 
761  for( auto &field : m_data ){
762  if( field.isEnabled() && field.getCodification() == VTKFormat::ASCII && field.getLocation() == VTKLocation::CELL ) {
763  str.seekg( position_insert);
764  readDataArray( str, field ) ;
765 
766  str.seekg( field.getPosition() ) ;
767  genericIO::copyUntilEOFInString( str, buffer, length );
768 
769  field.write( str ) ;

774  assert(false);
775  }
776 
777  str << std::endl ;
778  genericIO::flushBINARY( str, buffer, length) ;
779 
780  delete [] buffer ;
781  }
782  }

784  for( auto &field : m_geometry ){
785  if( field.isEnabled() && field.getCodification() == VTKFormat::ASCII ) {
786  str.seekg( position_insert);
787  readDataArray( str, field ) ;

794

797  assert(false);
798  }
799 
800  str << std::endl ;
801  genericIO::flushBINARY( str, buffer, length) ;
802 
803  delete [] buffer ;
804  }
805  }
806 
807  // Position the stream at the beginning of the stream
808  str.seekg(0, std::ios::beg);
809 
810  }
811 
812  { // Write Appended
813 
814  char c_;
815  std::string line ;
816  std::fstream::pos_type position_appended ;
817  std::fstream::pos_type position_before_write ;
818 
819  //Go to the initial position of the appended section
820  while( getline(str, line) && (! bitpit::utils::string::keywordInString( line, "<AppendedData")) ){}
821 
822  str >> c_;
823  while( c_ != '_') str >> c_;
824 
825  position_insert = str.tellg();
826  genericIO::copyUntilEOFInString( str, buffer, length );
827 
828  str.close();
829  str.clear();
830 
831 
832  //Reopening in binary mode
833  str.open( m_fh.getPath( ), std::ios::out | std::ios::in | std::ios::binary);
834  str.seekg( position_insert) ;
835 
836  //str.open( "data.dat", std::ios::out | std::ios::binary);
837 
838  //Writing first point data then cell data
839  for( auto &field : m_data ){
840  if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED && field.getLocation() == VTKLocation::POINT ) {
841  if( getHeaderType() == "UInt32"){

942  }

962  << " Vectors=" << vectors.str()
