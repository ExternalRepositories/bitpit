206 template<typename T>
207 ProxyVector<T>::ProxyVector(const ProxyVector &other)
208 {
209  // Copy the elements
210  std::vector<T_no_cv> new_storage(*(other.m_storage));
211 
212  // Assign the new memory to the object
213  m_storage = std::unique_ptr<std::vector<T_no_cv>>(new std::vector<T_no_cv>(std::move(new_storage)));
214  if (other.m_data == other.m_storage->data()) {
215  m_data = m_storage->data();
216  } else {
217  m_data = other.m_data;
