145  m_ghostExchangeSources(other.m_ghostExchangeSources)

148  // Register the patch
149  patch::manager().registerPatch(this);
150 
151  // Update the VTK streamer
152  //
153  // The pointer to VTK streamers are copied, if there are pointer to the
154  // original object they have to be replace with a pointer to this object.
155  std::vector<std::string> streamedGeomFields;
156  streamedGeomFields.reserve(m_vtk.getGeomDataCount());
157  for (auto itr = m_vtk.getGeomDataBegin(); itr != m_vtk.getGeomDataEnd(); ++itr) {
158  const VTKField &field = *itr;
159  if (&field.getStreamer() != &other) {
160  continue;
161  }
162 
163  streamedGeomFields.push_back(field.getName());
164  }
165 
166  for (const std::string &name : streamedGeomFields) {
167  const VTKField &field = *(m_vtk.findGeomData(name));
168  VTKField updatedField(field);
169  updatedField.setStreamer(*this);
170 
171  m_vtk.setGeomData(std::move(updatedField));
172  }
173 
174  std::vector<std::string> streamedDataFields;
175  streamedDataFields.reserve(m_vtk.getDataCount());
176  for (auto itr = m_vtk.getDataBegin(); itr != m_vtk.getDataEnd(); ++itr) {
177  const VTKField &field = *itr;
178  if (&field.getStreamer() != &other) {
179  continue;
180  }
181 
182  streamedDataFields.push_back(field.getName());
183  }
184 
185  for (const std::string &name : streamedDataFields) {
186  const VTKField &field = *(m_vtk.findData(name));
187  VTKField updatedField(field);
188  updatedField.setStreamer(*this);
189 
190  m_vtk.removeData(field.getName());
191  m_vtk.addData(std::move(updatedField));
192  }
193 
194 #if BITPIT_ENABLE_MPI==1
195  // Set the communicator
196  MPI_Comm communicator = other.getCommunicator();
197  if (communicator != MPI_COMM_NULL) {
198  setCommunicator(communicator);
199  }
200 #endif
201 }

310 std::vector<adaption::Info> PatchKernel::update(bool trackAdaption, bool squeezeStorage)
311 {
312  std::vector<adaption::Info> updateInfo;
313 
314  // Check if there are pending changes
315  bool spawnNeeed = (getSpawnStatus() == SPAWN_NEEDED);
316  bool adaptionDirty = (getAdaptionStatus(true) == ADAPTION_DIRTY);
317  bool boundingBoxDirty = isBoundingBoxDirty();
318 
319  bool pendingChanges = (spawnNeeed || adaptionDirty || boundingBoxDirty);
320  if (!pendingChanges) {
321  return updateInfo;
322  }
323 
324  // Spawn
325  if (spawnNeeed) {
326  mergeAdaptionInfo(spawn(trackAdaption), updateInfo);
327  }
328 
329  // Adaption
330  if (adaptionDirty) {
331  mergeAdaptionInfo(adaption(trackAdaption, squeezeStorage), updateInfo);
332  }
333 
334  // Update bounding box
335  //
336  // Previous updates may already have updated the bounding box, before
337  // doing the update re-check if it is still dirty.
338  boundingBoxDirty = isBoundingBoxDirty();
339  if (boundingBoxDirty) {
340  updateBoundingBox();
341  }
342 
343  return updateInfo;
344 }

353 std::vector<adaption::Info> PatchKernel::spawn(bool trackSpawn)
354 {
355  std::vector<adaption::Info> spawnInfo;
356 
357 #if BITPIT_ENABLE_MPI==1
358  // This is a collevtive operation and should be called by all processes
359  if (isCommunicatorSet()) {
360  const auto &communicator = getCommunicator();
361  MPI_Barrier(communicator);
362  }
363 #endif
364 
365  // Check spawn status
366  SpawnStatus spawnStatus = getSpawnStatus();
367  if (spawnStatus == SPAWN_UNNEEDED || spawnStatus == SPAWN_DONE) {
368  return spawnInfo;
369  }
370 
371  // Begin patch alteration
372  beginAlteration();
373 
374  // Alter patch
375  spawnInfo = _spawn(trackSpawn);
376 
377  // End patch alteration
378  endAlteration(true);
379 
380  // Spwan is done
381  setSpawnStatus(SPAWN_DONE);
382 
383  // Done
384  return spawnInfo;
385 }

744 void PatchKernel::write(VTKWriteMode mode)
745 {
746  // Set VTK targets
747  long vtkCellCount = 0;
748  if (m_vtkWriteTarget == WRITE_TARGET_CELLS_ALL) {
749  m_vtkCellRange = CellConstRange(cellConstBegin(), cellConstEnd());
750 
751  vtkCellCount = getCellCount();
752 #if BITPIT_ENABLE_MPI==1
753  } else if (m_vtkWriteTarget == WRITE_TARGET_CELLS_INTERNAL) {
754  m_vtkCellRange = CellConstRange(internalConstBegin(), internalConstEnd());
755 
756  vtkCellCount = getInternalCount();
757 #endif
758  }
759 
760  // Set the dimensinos of the mesh
761  PiercedStorage<long, long> vertexWriteFlag(1, &m_vertices);
762  vertexWriteFlag.fill(false);
763 
764  bool vtkFaceStreamNeeded = false;
765  for (const Cell &cell : m_cells) {
766  if (cell.getDimension() > 2 && !cell.hasInfo()) {
767  vtkFaceStreamNeeded = true;
768  break;
769  }
770  }
771 
772  long vtkConnectSize = 0;
773  long vtkFaceStreamSize = 0;
774  for (const Cell &cell : getVTKCellWriteRange()) {
775  ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
776  const int nCellVertices = cellVertexIds.size();
777  for (int k = 0; k < nCellVertices; ++k) {
778  long vertexId = cellVertexIds[k];
779  vertexWriteFlag.at(vertexId) = true;
780  }
781 
782  vtkConnectSize += nCellVertices;
783  if (vtkFaceStreamNeeded) {
784  if (cell.getDimension() <= 2 || cell.hasInfo()) {
785  vtkFaceStreamSize += 1;
786  } else {
787  vtkFaceStreamSize += cell.getFaceStreamSize();
788  }
789  }
790  }
791 
792  int vtkVertexCount = 0;
793  m_vtkVertexMap.unsetKernel(true);
794  m_vtkVertexMap.setStaticKernel(&m_vertices);
795  for (VertexConstIterator itr = vertexConstBegin(); itr != vertexConstEnd(); ++itr) {
796  std::size_t vertexRawId = itr.getRawIndex();
797  if (vertexWriteFlag.rawAt(vertexRawId)) {
798  m_vtkVertexMap.rawAt(vertexRawId) = vtkVertexCount++;
799  } else {
800  m_vtkVertexMap.rawAt(vertexRawId) = Vertex::NULL_ID;
801  }
802  }
803 
804  m_vtk.setDimensions(vtkCellCount, vtkVertexCount, vtkConnectSize, vtkFaceStreamSize);
805 
806  // Write the mesh
807  m_vtk.write(mode);
808 }
