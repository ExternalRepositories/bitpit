405  inline bool operator() (std::size_t pos_x, std::size_t pos_y)
406  {
407  id_t id_x = m_ids[pos_x];
408  id_t id_y = m_ids[pos_y];
409 
410  if (id_x >= 0 && id_y < 0) {
411  return true;
412  } else if (id_x < 0 && id_y >= 0) {
413  return false;
414  } else if (id_x >= 0) {
415  return (id_x < id_y);
416  } else {
417  return (id_x > id_y);
418  }
419  }
