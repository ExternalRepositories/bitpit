728 template<typename value_t, typename id_t>
729 void PiercedVector<value_t, id_t>::swap(PiercedVector &x) noexcept
730 {
731  // The swap will swap also the slave-master information. This is not what
732  // we want, therefore the two pierced storage will be unregistered and the
733  // registered again after the swap. When the kernel is unset the storage
734  // can't be clear, otherwise its contents will be lost.
735  PiercedVectorStorage<value_t, id_t>::detachKernel();
736  x.PiercedVectorStorage<value_t, id_t>::detachKernel();
737 
738  // Swap kernel data
739  PiercedVectorKernel<id_t>::swap(x);
740 
741  // Swap storage data
742  PiercedVectorStorage<value_t, id_t>::swap(x);
743 
744  // Re-register the storages
745  PiercedVectorStorage<value_t, id_t>::setDynamicKernel(this, PiercedVectorKernel<id_t>::SYNC_MODE_DISABLED);
746  x.PiercedVectorStorage<value_t, id_t>::setDynamicKernel(&x, PiercedVectorKernel<id_t>::SYNC_MODE_DISABLED);
747 }

813 template<typename value_t, typename id_t>
814 typename PiercedVector<value_t, id_t>::iterator PiercedVector<value_t, id_t>::reclaimValue(const FillAction &action)
815 {
816  std::size_t pos = action.info[PiercedSyncAction::INFO_POS];
817  switch (static_cast<typename FillAction::FillActionType>(action.type)) {
818 
819  case FillAction::TYPE_OVERWRITE:
820  {
821  // Nothing to do
822  break;
823  }
824 
825  case FillAction::TYPE_INSERT:
826  {
827  // Since we are increasing the sotrage by an element at the time
828  // calling a reserve will hurt performance badly because this will
829  // prevent the automatic reallocation of the storage.
830  PiercedVectorStorage<value_t, id_t>::rawEmplace(pos);
831  break;
832  }
833 
834  case FillAction::TYPE_APPEND:
835  {
836  // Since we are increasing the sotrage by an element at the time
837  // calling a reserve will hurt performance badly because this will
838  // prevent the automatic reallocation of the storage.
839  PiercedVectorStorage<value_t, id_t>::rawEmplaceBack();
840  pos = PiercedVectorKernel<id_t>::getLastUsedPos();
841  break;
842  }
843 
844  default:
845  {
846  BITPIT_UNREACHABLE("This action is not handled");
847  break;
848  }
849 
850  }
851 
852  // Return the iterator to the position where the element was inserted
853  return PiercedVectorStorage<value_t, id_t>::rawFind(pos);
854 }
