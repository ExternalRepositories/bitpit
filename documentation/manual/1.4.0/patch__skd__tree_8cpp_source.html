58 void SkdPatchInfo::buildCache()
59 {
60  const PiercedVector<Cell> &cells = m_patch->getCells();
61 
62  m_cellBoxes = std::unique_ptr<BoxCache>(new BoxCache(2, &(m_patch->getCells())));
63  m_cellCentroids = std::unique_ptr<BoxCache>(new CentroidCache(1, &(m_patch->getCells())));
64  for (auto itr = m_patch->internalConstBegin(); itr != m_patch->internalConstEnd(); ++itr) {
65  std::size_t rawCellId = itr.getRawIndex();
66 
67  // Cell info
68  const Cell &cell = cells.rawAt(rawCellId);
69  int nCellVertices = cell.getVertexCount();
70  const long *cellConnect = cell.getConnect();
71 
72  // Bounding box
73  std::array<double, 3> &cellBoxMin = m_cellBoxes->rawAt(rawCellId, 0);
74  std::array<double, 3> &cellBoxMax = m_cellBoxes->rawAt(rawCellId, 1);
75  std::array<double, 3> &cellCentroid = m_cellCentroids->rawAt(rawCellId, 0);
76 
77  cellBoxMin = m_patch->getVertexCoords(cellConnect[0]);
78  cellBoxMax = cellBoxMin;
79  cellCentroid = cellBoxMin;
80  for (int i = 1; i < nCellVertices; ++i) {
81  const std::array<double, 3> &coords = m_patch->getVertexCoords(cellConnect[i]);
82  for (int d = 0; d < 3; ++d) {
83  cellBoxMin[d] = std::min(coords[d], cellBoxMin[d]);
84  cellBoxMax[d] = std::max(coords[d], cellBoxMax[d]);
85  cellCentroid[d] += coords[d];
86  }
87  }
88  cellCentroid /= double(nCellVertices);
89  }
90 }

218 void SkdNode::initializeBoundingBox()
219 {
220  const std::vector<std::size_t> &cellRawIds = m_patchInfo->getCellRawIds();
221 
222  // Evaluate the bounding box
223  m_boxMin = m_patchInfo->getCachedBoxMin(cellRawIds[m_cellRangeBegin]);
224  m_boxMax = m_patchInfo->getCachedBoxMax(cellRawIds[m_cellRangeBegin]);
225  for (std::size_t n = m_cellRangeBegin + 1; n < m_cellRangeEnd; n++) {
226  const std::size_t rawCellId = cellRawIds[n];
227  const std::array<double, 3> &cellBoxMin = m_patchInfo->getCachedBoxMin(rawCellId);
228  const std::array<double, 3> &cellBoxMax = m_patchInfo->getCachedBoxMax(rawCellId);
229  for (int d = 0; d < 3; ++d) {
230  m_boxMin[d] = std::min(cellBoxMin[d], m_boxMin[d]);
231  m_boxMax[d] = std::max(cellBoxMax[d], m_boxMax[d]);
232  }
233  }
234 
235  // Inlfate the bounding box by a small epsilon
236  //
237  // The small inflation allows us to test intersection of a query
238  // against a safety box and be assured that non-intersection of
239  // with the box implies non-intersection with all the contents
240  // in the box, even when computations are susceptible to round-off
241  // error.
242  double tolerance = m_patchInfo->getPatch().getTol();
243 
244  m_boxMin -= tolerance;
245  m_boxMax += tolerance;
246 }

449 void SkdNode::findPointClosestCell(const std::array<double, 3> &point,
450  long *id, double *distance) const
451 {
452  *id = Cell::NULL_ID;
453  *distance = std::numeric_limits<double>::max();
454  if (getCellCount() == 0) {
455  return;
456  }
457 
458  const PatchKernel &patch = m_patchInfo->getPatch();
459  const PiercedVector<Cell> &cells = patch.getCells();
460  const std::vector<std::size_t> &cellRawIds = m_patchInfo->getCellRawIds();
461 
462  for (std::size_t n = m_cellRangeBegin; n < m_cellRangeEnd; n++) {
463  std::size_t cellRawId = cellRawIds[n];
464  const Cell &cell = cells.rawAt(cellRawId);
465  long cellId = cell.getId();
466  const long *cellConnect = cell.getConnect();
467 
468  // Evaluate the distance from the cell
469  double cellDistance;
470 
471  ElementType cellType = cell.getType();
472  switch (cellType) {
473 
474  case ElementType::LINE:
475  {
476  const std::array<double, 3> &vertexCoords_0 = patch.getVertexCoords(cellConnect[0]);
477  const std::array<double, 3> &vertexCoords_1 = patch.getVertexCoords(cellConnect[1]);
478 
479  cellDistance = CGElem::distancePointSegment(point, vertexCoords_0, vertexCoords_1);
480  break;
481  }
482 
483  case ElementType::TRIANGLE:
484  {
485  const std::array<double, 3> &vertexCoords_0 = patch.getVertexCoords(cellConnect[0]);
486  const std::array<double, 3> &vertexCoords_1 = patch.getVertexCoords(cellConnect[1]);
487  const std::array<double, 3> &vertexCoords_2 = patch.getVertexCoords(cellConnect[2]);
488 
489  cellDistance = CGElem::distancePointTriangle(point, vertexCoords_0, vertexCoords_1, vertexCoords_2);
490  break;
491  }
492 
493  case ElementType::PIXEL:
494  case ElementType::QUAD:
495  {
496  std::vector<std::array<double, 3>> vertexCoords;
497  int nVertices = cell.getVertexCount();
498  vertexCoords.resize(nVertices);
499  for (int i = 0; i < nVertices; ++i) {
500  vertexCoords[i] = patch.getVertexCoords(cellConnect[i]);
501  }
502 
503  cellDistance = CGElem::distancePointPolygon(point, vertexCoords);
504  break;
505  }
506 
507  default:
508  throw std::runtime_error ("Type of cell not supported.");
509 
510  }
511 
512  // Update the distance
513  if (cellDistance < *distance) {
514  *distance = cellDistance;
515  *id = cellId;
516  }
517  }
518 }

554 bool SkdNode::boxIntersectsSphere(const std::array<double, 3> &center, double radius) const
555 {
556  // If the box contains the center it will also intersect the sphere
557  if (boxContainsPoint(center, 0.)) {
558  return true;
559  }
560 
561  // Check if the distance between the center and its projection on
562  // the bounding box is smaller than the radius of the sphere.
563  std::array<double, 3> delta;
564  for (int d = 0; d < 3; d++) {
565  delta[d] = std::min(std::max(center[d], m_boxMin[d]), m_boxMax[d]) - center[d];
566  }
567  double distance = dotProduct(delta, delta);
568 
569  return (distance < radius * radius);
570 }

662 void PatchSkdTree::build(int leafCapacity)
663 {
664  const PatchKernel &patch = m_patchInfo.getPatch();
665 
666  // Clear existing tree
667  clear();
668 
669  // Set leaf size
670  setLeafCapacity(leafCapacity);
671 
672  // Initialize list of cell raw ids
673  std::size_t nCells = patch.getInternalCount();
674  if (m_cellRawIds.size() != nCells) {
675  m_cellRawIds.resize(nCells);
676  }
677 
678  std::size_t k = 0;
679  for (auto itr = patch.internalConstBegin(); itr != patch.internalConstEnd(); ++itr) {
680  m_cellRawIds[k++] = itr.getRawIndex();
681  }
682 
683  // Build patch cache
684  m_patchInfo.buildCache();
685 
686  // Initialize node list
687  m_nodes.reserve(1.5 * nCells / getLeafCapacity());
688 
689  // Create the root
690  m_nodes.emplace_back(&m_patchInfo, 0, nCells);
691 
692  // Create the tree
693  std::queue<std::size_t> nodeStack;
694  nodeStack.push(0);
695  while (!nodeStack.empty()) {
696  std::size_t nodeId = nodeStack.front();
697  nodeStack.pop();
698 
699  // Create the children of the node
700  createChildren(nodeId);
701 
702  // Add the newly created childrend to the stack
703  const SkdNode &node = getNode(nodeId);
704  for (int i = SkdNode::CHILD_BEGIN; i != SkdNode::CHILD_END; ++i) {
705  std::size_t childId = node.getChildId(static_cast<SkdNode::ChildLocation>(i));
706  if (childId != SkdNode::NULL_ID) {
707  nodeStack.push(childId);
708  }
709  }
710  }
711 
712  // Patch cache is no longer needed
713  m_patchInfo.destroyCache();
714 }

817 void PatchSkdTree::createChildren(std::size_t parentId)
818 {
819  SkdNode &parent = _getNode(parentId);
820 
821  // Check if the parent is a leaf.
822  if (parent.getCellCount() <= m_leafCapacity) {
823  ++m_nLeafs;
824  return;
825  }
826 
827  // Evaluate the preferred direction along which elements will be split.
828  //
829  // The elements will be split along a plane normal to the direction
830  // for which the bounding box has the maximum length.
831  const std::array<double, 3> &parentBoxMin = parent.getBoxMin();
832  const std::array<double, 3> &parentBoxMax = parent.getBoxMax();
833 
834  int largerDirection = 0;
835  double boxMaximumLength = parentBoxMax[largerDirection] - parentBoxMin[largerDirection];
836  for (int d = 1; d < 3; ++d) {
837  double length = parentBoxMax[d] - parentBoxMin[d];
838  if (length > boxMaximumLength) {
839  largerDirection = d;
840  boxMaximumLength = length;
841  }
842  }
843 
844  // Split the elements.
845  //
846  // Preferred direction is tried first, if the split along this direction
847  // will produce and empty child (e.g., all the cell's box centroids have
848  // the same coordinate along the split direction and therefore all the
849  // cells would be assigned to the left chiled) the split will be performed
850  // along one of the other directions.
851  std::array<double, 3> parentWeightedCentroid = parent.evalBoxWeightedMean();
852  for (int d = 0; d < 3; ++d) {
853  // Update the split direction
854  int splitDirection = (largerDirection + d) % 3;
855 
856  // Get the threshold for the split
857  double splitThreshold = parentWeightedCentroid[splitDirection];
858 
859  // Order the elements
860  //
861  // All the elements with a centroid coordinate less or equal than the
862  // threshold will be assigned to the left child, the others will be
863  // assigned to the right child.
864  std::size_t leftBegin = parent.m_cellRangeBegin;
865  std::size_t leftEnd = parent.m_cellRangeEnd;
866  std::size_t rightBegin = parent.m_cellRangeBegin;
867  std::size_t rightEnd = parent.m_cellRangeEnd;
868  while (true) {
869  // Update the right begin
870  while (rightBegin != leftEnd && m_patchInfo.getCachedCentroid(m_cellRawIds[rightBegin])[splitDirection] <= splitThreshold) {
871  rightBegin++;
872  }
873 
874  // Update the left end
875  while (rightBegin != leftEnd && m_patchInfo.getCachedCentroid(m_cellRawIds[leftEnd - 1])[splitDirection] > splitThreshold) {
876  leftEnd--;
877  }
878 
879  // If all the elements are in the right position we can exit
880  if (rightBegin == leftEnd) {
881  break;
882  }
883 
884  // If left end and and right begin are not equal, that the two ids
885  // point to misplaced elements. Swap the elements, advance the ids
886  // and continue iterating.
887  std::iter_swap(m_cellRawIds.begin() + (leftEnd - 1), m_cellRawIds.begin() + rightBegin);
888  }
889 
890  if (leftEnd <= leftBegin || rightEnd <= rightBegin) {
891  continue;
892  }
893 
894  // Create the left child
895  long leftId = m_nodes.size();
896  m_nodes.emplace_back(&m_patchInfo, leftBegin, leftEnd);
897  m_nodes[parentId].m_children[static_cast<std::size_t>(SkdNode::CHILD_LEFT)] = leftId;
898 
899  // Create the right child
900  long rightId = m_nodes.size();
901  m_nodes.emplace_back(&m_patchInfo, rightBegin, rightEnd);
902  m_nodes[parentId].m_children[static_cast<std::size_t>(SkdNode::CHILD_RIGHT)] = rightId;
903 
904  // Done
905  return;
906  }
907 
908  throw std::runtime_error("Unable to create the children.");
909 }
