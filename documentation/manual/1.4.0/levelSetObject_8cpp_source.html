204 LevelSetIntersectionStatus LevelSetObject::intersectSurface(const long &i, LevelSetIntersectionMode mode) const{
205 
206  double incircle, circumcircle;
207 
208  switch(mode){
209  case LevelSetIntersectionMode::FAST_GUARANTEE_TRUE:
210  incircle = m_kernelPtr->computeCellIncircle(i) ;
211  if(std::abs(getLS(i)) <= incircle){
212  return LevelSetIntersectionStatus::TRUE;
213  } else {
214  return LevelSetIntersectionStatus::FALSE;
215  }
216 
217  break;
218 
219  case LevelSetIntersectionMode::FAST_GUARANTEE_FALSE:
220  circumcircle = m_kernelPtr->computeCellCircumcircle(i) ;
221  if(std::abs(getLS(i)) > circumcircle){
222  return LevelSetIntersectionStatus::FALSE;
223  } else {
224  return LevelSetIntersectionStatus::TRUE;
225  }
226 
227  break;
228 
229  case LevelSetIntersectionMode::FAST_FUZZY:
230  circumcircle = m_kernelPtr->computeCellCircumcircle(i) ;
231  if(std::abs(getLS(i)) > circumcircle){
232  return LevelSetIntersectionStatus::FALSE;
233  }
234 
235  incircle = m_kernelPtr->computeCellIncircle(i) ;
236  if(std::abs(getLS(i)) <= incircle){
237  return LevelSetIntersectionStatus::TRUE;
238  }
239 
240  return LevelSetIntersectionStatus::CLOSE;
241 
242  break;
243 
244  case LevelSetIntersectionMode::ACCURATE:
245  circumcircle = m_kernelPtr->computeCellCircumcircle(i) ;
246  if(std::abs(getLS(i)) > circumcircle){
247  return LevelSetIntersectionStatus::FALSE;
248  }
249 
250  incircle = m_kernelPtr->computeCellIncircle(i) ;
251  if(std::abs(getLS(i)) <= incircle){
252  return LevelSetIntersectionStatus::TRUE;
253  }
254 
255  std::array<double,3> root = computeProjectionPoint(i);
256  std::array<double,3> normal = getNormal(i);
257  if( m_kernelPtr->intersectCellPlane(i,root,normal) ){
258  return LevelSetIntersectionStatus::TRUE;
259  } else {
260  return LevelSetIntersectionStatus::FALSE;
261  }
262 
263  break;
264  }
265 
266  BITPIT_UNREACHABLE("cannot reach");
267 
268 }

399 void LevelSetObject::communicate( const std::unordered_map<int,std::vector<long>> &sendList,
400  const std::unordered_map<int,std::vector<long>> &recvList,
401  std::vector<adaption::Info> const *mapper){
402 
403  if (!m_kernelPtr->getMesh()->isPartitioned()) {
404  return;
405  }
406 
407  // Create a data communicator
408  MPI_Comm meshComm = m_kernelPtr->getCommunicator() ;
409  DataCommunicator dataCommunicator(meshComm) ;
410 
411  // Fill the send buffer with the content from the LevelSetObject base
412  // class and the specific derived class.
413  for (const auto entry : sendList) {
414  // Create an empty send
415  int rank = entry.first;
416  dataCommunicator.setSend(rank, 0);
417 
418  // Write data in the buffer
419  SendBuffer &buffer = dataCommunicator.getSendBuffer(rank);
420  writeCommunicationBuffer(entry.second, buffer);
421  }
422 
423  // Discover the receives
424  dataCommunicator.discoverRecvs();
425 
426  // Start the sends
427  dataCommunicator.startAllRecvs();
428 
429  // In case of mesh adaption, the ids of the moved items must be freed
430  // in order to avoid conflicts when ids are recycled
431  if (mapper) {
432  clearAfterMeshAdaption(*mapper);
433  }
434 
435  // Start the sends
436  dataCommunicator.startAllSends();
437 
438  // Check which data communications have arrived. For those which are
439  // available start reading the databuffer into the data structure of
440  // LevelSetObject and its derived classes.
441  int nCompletedRecvs = 0;
442  while (nCompletedRecvs < dataCommunicator.getRecvCount()) {
443  int rank = dataCommunicator.waitAnyRecv();
444 
445  RecvBuffer &dataBuffer = dataCommunicator.getRecvBuffer(rank);
446  readCommunicationBuffer(recvList.at(rank), dataBuffer);
447 
448  ++nCompletedRecvs;
449  }
450 
451  dataCommunicator.waitAllSends();
452  dataCommunicator.finalize();
453 }
