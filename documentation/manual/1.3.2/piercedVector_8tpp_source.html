764 template<typename value_t, typename id_t>
765 void PiercedVector<value_t, id_t>::resize(std::size_t n)
766 {
767  // If the size of the vector is already the requested size
768  // there is nothing to do.
769  if (n == size()) {
770  return;
771  }
772 
773  // A request for a size equal to 0 is equivalent to a clear.
774  if (n == 0) {
775  clear();
776  return;
777  }
778 
779  // If the requested size is greater that the current size we
780  // may need to reserve space in the storage to allow the
781  // container to reach the requested size.
782  if (n > size()) {
783  reserve(n);
784  return;
785  }
786 
787  // If the requested size is smaller that the current size
788  // we need to perform a real resize.
789 
790  // Flush holes
791  holesFlush();
792 
793  // Find the id of the last element
794  id_t last_id = getSizeMarker(n - 1);
795 
796  // Find the updated last position
797  std::size_t updated_last_pos = getPosFromId(last_id);
798 
799  // Update the last position
800  updateLastUsedPos(updated_last_pos);
801 }

806 template<typename value_t, typename id_t>
807 void PiercedVector<value_t, id_t>::sort()
808 {
809  // Squeeze the container
810  squeeze();
811 
812  // The container has been squeezed, there are no holes
813  size_t containerSize = size();
814 
815  // Evaluates the permutations
816  //
817  // The permutation for ids and values are the same, however the reored
818  // function will destroy the permutation on output, so two different
819  // permutations are needed.
820  std::vector<std::size_t> id_permutation;
821  id_permutation.resize(containerSize);
822  for (size_t i = 0; i < containerSize; ++i) {
823  id_permutation[i] = i;
824  }
825  std::sort(id_permutation.begin(), id_permutation.end(), idLess(m_ids));
826 
827  std::vector<std::size_t> value_permutation(id_permutation);
828 
829  // Sort the container
830  utils::reorderVector<id_t>(id_permutation, m_ids, containerSize);
831  utils::reorderVector<value_t>(value_permutation, m_v, containerSize);
832 }

844 template<typename value_t, typename id_t>
845 void PiercedVector<value_t, id_t>::squeeze()
846 {
847  // Flush changes
848  flush();
849 
850  // Compact the vector
851  std::size_t nHoles = holesCount();
852  if (nHoles != 0) {
853  // Move the elements
854  std::size_t firstPosToUpdate;
855  if (m_first_pos == 0) {
856  firstPosToUpdate = *(m_holes_regular_end - 1);
857  } else {
858  firstPosToUpdate = 0;
859  }
860 
861  std::size_t offset = 0;
862  for (std::size_t pos = firstPosToUpdate; pos <= m_last_pos; pos++) {
863  if (offset < nHoles && *(m_holes_regular_end - offset - 1) == pos) {
864  ++offset;
865  continue;
866  }
867 
868  id_t id = m_ids[pos];
869  std::size_t updatedPos = pos - offset;
870 
871  setPosId(updatedPos, id);
872  setEmptyPosId(pos, pos + 1);
873  m_v[updatedPos] = std::move(m_v[pos]);
874  }
875 
876  // Clear the holes
877  holesClear();
878 
879  // Reset first and last counters
880  updateFirstUsedPos(0);
881  updateLastUsedPos(size() - 1);
882  }
883 
884  // Shrink to fit
885  m_ids.shrink_to_fit();
886  m_v.shrink_to_fit();
887 }

1230 template<typename value_t, typename id_t>
1231 id_t PiercedVector<value_t, id_t>::getSizeMarker(const size_t &targetSize, const id_t &fallback)
1232 {

1234  // size is equal to the size minus one we return the last element,
1235  // if the target size is greater or equal the current container size
1236  // we return the fallback value.
1237  if (targetSize >= size()) {
1238  return fallback;
1239  } else if (targetSize == 0) {
1240  return m_ids[m_first_pos];
1241  } else if (targetSize == (size() - 1)) {
1242  return m_ids[m_last_pos];
1243  }
1244 
1245  // Sort the holes
1246  holesSortRegular();
1247  holesSortPending();
1248

1250  // requeste number of element.

1252  hole_iterator pending_hole_itr = m_holes_pending_end;

1255  std::size_t markerPos = targetSize;

1262  // current marker position
1263  if (regular_hole_itr != m_holes_regular_begin) {

1265  regular_hole_itr = std::upper_bound(m_holes_regular_begin, regular_hole_itr, markerPos, std::greater<std::size_t>());

1267  }

1270  hole_iterator itr_previous = pending_hole_itr;
1271  pending_hole_itr = std::upper_bound(m_holes_pending_begin, pending_hole_itr, markerPos, std::greater<std::size_t>());

1278  // we update the marker and we continue iterating

1283  markerPos += targetSize - markerSize;
1284  }
