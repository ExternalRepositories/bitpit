239 double LevelSetCachedObject::_computeSizeNarrowBand( LevelSetOctree *visitee ){
240 
241  VolOctree& mesh = *(visitee->getOctreeMesh());
242  double RSearch(0.) ;
243 
244  double size;
245 
246  bool flagged ;
247  std::vector<bool> nearPoint ;
248 
249  int i, level(100) ;
250  uint8_t j0(0), j1( pow(2,mesh.getDimension())-1) ;
251 
252  std::array<double,3> octrBB0, octrBB1, triBB0, triBB1, C0, C1 ;
253 
254  // finest cell in octree
255  size = mesh.getTree().getLocalMinSize();
256 
257  visitee->getMesh()->getBoundingBox(octrBB0, octrBB1) ;
258  getBoundingBox(triBB0, triBB1) ;
259 
260  if( CGElem::intersectBoxBox(octrBB0,octrBB1,triBB0,triBB1,C0,C1) ) { //intersect two Bounding Boxes around geometry and local grid
261 
262  // snap bounding box to grid and create cartesian grid
263  std::array<int,3> nc ;
264 
265  C0 -= size ;
266  C1 += size ;
267 
268  for( i=0; i<mesh.getDimension(); ++i){
269  C0[i] = octrBB0[i] + size * (int) ( ( C0[i] - octrBB0[i] ) / size ) ;
270  C1[i] = octrBB0[i] + size * ( (int) ( ( C1[i] - octrBB0[i] ) / size ) +1 ) ;
271 
272  nc[i] = round( ( C1[i] - C0[i] ) /size ) ;
273  }
274 
275  // calculate LS on cartesian mesh and calculate RSearch by finding largest cell throughout flagged cartesian cells
276  VolCartesian cmesh( 0, mesh.getDimension(), C0, C1-C0, nc ) ;
277  LevelSet auxLS ;
278 
279  auxLS.setMesh( &cmesh ) ;
280  int objectId = auxLS.addObject( std::unique_ptr<LevelSetObject>(clone()) ) ;
281 
282  auxLS.setSign(false) ;
283  auxLS.compute( ) ;
284 
285  const LevelSetObject &obj = auxLS.getObject(objectId);
286 
287  std::array<int,3> i0;
288  std::array<int,3> i1;
289  int _i, _j, _k, index, twoDAdjust;
290 
291  twoDAdjust = 3-mesh.getDimension() ;
292 
293  for( const auto &cell : mesh.getCells() ){
294 
295  const long int* conn = cell.getConnect() ;
296 
297  C0 = mesh.getVertexCoords(conn[j0]) ;
298  C1 = mesh.getVertexCoords(conn[j1]) ;
299 
300  i0 = cmesh.locateClosestVertexCartesian(C0);
301  i1 = cmesh.locateClosestVertexCartesian(C1);
302 
303  i1[2] += twoDAdjust ;
304 
305  flagged = false ;
306 
307  for( _k=i0[2]; _k<i1[2]; ++_k){
308  for( _j=i0[1]; _j<i1[1]; ++_j){
309  for( _i=i0[0]; _i<i1[0]; ++_i){
310 
311  index = cmesh.getCellLinearId( _i, _j, _k) ;
312  flagged = flagged || obj.isInNarrowBand(index) ;
313 
314  }
315  }
316  }
317 
318  if(flagged)
319  level = std::min( level, mesh.getCellLevel( cell.getId() ) ) ;
320 
321  }
322 
323  RSearch = visitee->computeRSearchFromLevel( level ) ;
324 
325  } //endif intersect
326 
327 # if BITPIT_ENABLE_MPI
328  if( assureMPI() ){
329  double reducedRSearch ;
330  MPI_Comm meshComm = visitee->getCommunicator() ;
331  MPI_Allreduce( &RSearch, &reducedRSearch, 1, MPI_DOUBLE, MPI_MAX, meshComm );
332  RSearch = reducedRSearch ;
333  }
334 #endif
335 
336  return RSearch;
337 }

386 double LevelSetCachedObject::_updateSizeNarrowBand(LevelSetOctree *visitee, const std::vector<adaption::Info> &mapper){
387 
388  VolOctree const &mesh = *(visitee->getOctreeMesh());
389  double newRSearch(0.) ;
390 
391  bool coarseningInNarrowBand=false ;
392 
393  // assumes that LS information is relevant to OLD!!! grid
394  // scrrens old narrow band for coarsest elements
395 
396  //
397  // Get the bounding box of the objects
398  std::array<double, 3> objectsMinPoint;
399  std::array<double, 3> objectsMaxPoint;
400  getBoundingBox( objectsMinPoint, objectsMaxPoint ) ;
401 
402  //
403  // Cells in the narrow band
404  //
405  std::unordered_set<long> narrowBandCells;
406 
407  // First insert the new cells added to the narrow band
408  std::unordered_set<long> removedNBCells;
409  for ( auto & info : mapper ){
410  if( info.entity != adaption::Entity::ENTITY_CELL){
411  continue;
412  }
413 
414  bool parentInNarrowBand = false;
415  for ( auto & parent : info.previous){
416  if( isInNarrowBand(parent) ){
417  parentInNarrowBand = true;
418  if( info.type == adaption::Type::TYPE_COARSENING ){
419  coarseningInNarrowBand=true ;
420  }
421  break;
422  }
423  }
424 
425  if (parentInNarrowBand) {
426  for ( auto & parent : info.previous){
427  removedNBCells.insert(parent);
428  }
429 
430  for( auto &child : info.current){
431  narrowBandCells.insert(child);
432  }
433  }
434  }
435 
436  // Now add the cells that were in the narrow band befor and have not been
437  // updated
438  for (auto cell : mesh.getCells()) {
439  long id = cell.getId() ;
440  if( removedNBCells.count(id) > 0 || !isInNarrowBand(id) ) {
441  continue;
442  }
443 
444  narrowBandCells.insert(id);
445  }
446 
447  // Evaluate if the cells in the narrow band are inside the bounding box
448  // defined by the objects.
449  std::unordered_map<long, bool> isInsideObjectBox;
450  for ( long cellId : narrowBandCells ){
451  isInsideObjectBox.insert( { cellId, visitee->isCellInsideBoundingBox( cellId, objectsMinPoint, objectsMaxPoint ) } );
452  }
453 
454  //
455  // Get the miminum level in the narrow band
456  //
457  // We need to consider only the cells that are inside the bounding box
458  // defined by the objects, or the cells that have at least a neighbout
459  // inside the bounding box defined by the objects.
460  for ( long cellId : narrowBandCells ){
461  // Discard cells that are not in the bounding box
462  bool discardCell = ! isInsideObjectBox.at(cellId) ;
463  if ( discardCell ) {
464  for ( long neighId : mesh.findCellFaceNeighs(cellId) ) {
465  if ( narrowBandCells.count(neighId) == 0 ) {
466  continue;
467  }
468 
469  if (isInsideObjectBox.at(neighId)) {
470  discardCell = false ;
471  break ;
472  }
473  }
474  }
475 
476  if (discardCell) {
477  continue;
478  }
479 
480  // Update the level
481  newRSearch = std::max( newRSearch, visitee->computeRSearchFromCell(cellId) ) ;
482 
483  }
484 
485  if(coarseningInNarrowBand==false){
486  newRSearch = std::min( newRSearch, getSizeNarrowBand() ) ;
487  }
488 
489 # if BITPIT_ENABLE_MPI
490  if( assureMPI() ) {
491  double reducedRSearch ;
492  MPI_Comm meshComm = visitee->getCommunicator() ;
493  MPI_Allreduce( &newRSearch, &reducedRSearch, 1, MPI_DOUBLE, MPI_MAX, meshComm );
494  newRSearch = reducedRSearch ;
495  }
496 #endif
497 
498  return newRSearch;
499 
500 }

505 void LevelSetCachedObject::propagateSign() {
506 
507  VolumeKernel const &mesh = *(m_kernelPtr->getMesh()) ;
508 
509  // Save the bounding boxes of the object
510  std::array<double,3> boxMin;
511  std::array<double,3> boxMax;
512  getBoundingBox(boxMin, boxMax);
513 
514  // Identify the regions
515  //
516  // A region is a group of cells that share at leas a face. There are three
517  // type of regions: external, internal, narrowband, and unknown.
518  //
519  // An 'external' region is outside all bodies, an 'internal' region is
520  // inside of at least one body, a 'narrowband' region contains at least
521  // one cell in the narrow band, whereas it is not easly possible to find
522  // out the position of an 'unknown' region.
523  const int REGION_EXTERNAL = 0;
524  const int REGION_INTERNAL = 1;
525  const int REGION_NARROWBAND = 2;
526  const int REGION_UNKNOWN = 3;
527 
528  const std::vector<long> &cellIds = mesh.getCells().getIds();
529 
530  int nRegions = 0;
531  std::vector<int> regionType;
532  std::vector<std::unordered_set<long>> regionCellList;
533  std::unordered_set<long> alreadyAssigned;
534  for (long id : cellIds) {
535  // Skip the cell if is already assigned to a region
536  if (alreadyAssigned.count(id) != 0) {
537  continue;
538  }
539 
540  // Creat a new region
541  int region = nRegions++;
542  regionType.push_back(REGION_EXTERNAL);
543  regionCellList.emplace_back();
544 
545  // Find all the cells that belongs to the region
546  std::stack<long> seeds;
547  seeds.push(id);
548 
549  while (!seeds.empty()) {
550  long seed = seeds.top();
551  seeds.pop();
552 
553  // Seeds assigned to a region has already been processed
554  if (alreadyAssigned.count(seed) != 0) {
555  continue;
556  }
557 
558  // Assign the current region to the seed
559  regionCellList[region].insert(seed);
560 
561  // Detect the region type
562  if (regionType[region] != REGION_NARROWBAND) {
563  if (isInNarrowBand(seed)) {
564  regionType[region] = REGION_NARROWBAND;
565  } else if (regionType[region] != REGION_UNKNOWN) {
566  std::array<double, 3> centroid = mesh.evalCellCentroid(seed);
567 
568  bool isInside = true;
569  for (int i = 0; i < 3; ++i) {
570  if (centroid[i] < boxMin[i] || centroid[i] > boxMax[i]) {
571  isInside = false;
572  break;
573  }
574  }
575 
576  if (isInside) {
577  regionType[region] = REGION_UNKNOWN;
578  }
579  }
580  }
581 
582  // Add the unassigned neighboors to the seeds
583  for (long neigh : mesh.findCellFaceNeighs(seed)) {
584  if (alreadyAssigned.count(neigh) == 0) {
585  seeds.push(neigh);
586  }
587  }
588 
589  // The cell has been assigned to a region
590  alreadyAssigned.insert(seed);
591  }
592  }
593 
594  // According to the region type propagate the sign.
595  //
596  // If a region is 'external', all its cells have a positive levelset. If
597  // a region is 'narrowband' it is possible to propagate the sign of the
598  // narrowband to the whole region. All the cells of an 'unknown' region
599  // will have the same sign, but the value hasto be obtain from the
600  // neighbouring partitions.
601  for (int region = 0; region < nRegions; ++region) {
602  if (regionType[region] == REGION_EXTERNAL) {
603  assignSign(1, regionCellList[region]);
604  } else if (regionType[region] == REGION_NARROWBAND) {
605  // The seeds are the cells in the narrowband of the region
606  std::stack<long> seeds;
607  std::unordered_set<long> alreadyEvaluated;
608  for (long id : regionCellList[region]) {
609  if (!isInNarrowBand(id)) {
610  continue;
611  }
612 
613  seeds.push(id);
614  alreadyEvaluated.insert(id);
615  }
616 
617  // Propagate the sign
618  while (!seeds.empty()) {
619  long seed = seeds.top();
620  seeds.pop();
621 
622  // Discard seeds with a LS value equal to 0
623  double ls = getLS(seed);
624  if (utils::DoubleFloatingEqual()(std::abs(ls), (double) 0.)) {
625  continue;
626  }
627 
628  // Start filling the process list with the neighbours of the
629  // seeds that has not yet been evaluated.
630  //
631  // If a seed is surrounded only by items already evaluated,
632  // it can't propagate the sign to anyone.
633  std::stack<long> processList;
634  for (long neigh : mesh.findCellFaceNeighs(seed)) {
635  if (alreadyEvaluated.count(neigh) == 0) {
636  processList.push(neigh);
637  }
638  }
639 
640  if (processList.empty()) {
641  continue;
642  }
643 
644  // Get the sign of the seed
645  short seedSign = getSign(seed);
646 
647  // Propagate the sign
648  while (!processList.empty()) {
649  long id = processList.top();
650  processList.pop();
651 
652  // Get the value associated to the id
653  //
654  // A new value needs to be created only if the sign to
655  // propagate is different from the default sign.
656  PiercedVector<LevelSetInfo>::iterator infoItr = m_ls.find(id) ;
657  if( infoItr == m_ls.end() && seedSign != levelSetDefaults::SIGN ){
658  infoItr = m_ls.reclaim(id) ;
659  }
660 
661  // Update the value
662  if( infoItr != m_ls.end() ) {
663  (*infoItr).value = seedSign * levelSetDefaults::VALUE;
664  }
665 
666  // Add non-evaluated neighs to the process list
667  std::vector<long> neighs = mesh.findCellFaceNeighs( id ) ;
668  for (long neigh : neighs) {
669  if (alreadyEvaluated.count(neigh) == 0) {
670  processList.push(neigh);
671  }
672  }
673 
674  // The cell has been processeed
675  alreadyEvaluated.insert(id);
676  }
677 
678  // If all cells have been evaluated we can stop the propagation
679  if (alreadyEvaluated.size() == regionCellList[region].size()) {
680  break;
681  }
682  }
683  }
684  }
685 
686 #if BITPIT_ENABLE_MPI
687  // If the communicator is not set we can exit because the calculation
688  // is serial
689  if (!m_kernelPtr->isCommunicatorSet()) {
690  return;
691  }
692 
693  // If there is only one processor we can exit
694  int nProcs;
695  MPI_Comm_size(m_kernelPtr->getCommunicator(), &nProcs);
696  if (nProcs == 1) {
697  return;
698  }
699 
700  // Initialize the communicator for exchanging the sign among partitions
701  DataCommunicator dataCommunicator(m_kernelPtr->getCommunicator());
702  dataCommunicator.setTag(108) ;
703 
704  int sign;
705  size_t dataSize = sizeof(sign);
706 
707  // Set the receives
708  for (const auto entry : mesh.getGhostExchangeTargets()) {
709  const int rank = entry.first;
710  const auto &list = entry.second;
711 
712  dataCommunicator.setRecv(rank, list.size() * dataSize);
713  }
714 
715  // Set the sends
716  for (const auto entry : mesh.getGhostExchangeSources()) {
717  const int rank = entry.first;
718  auto &list = entry.second;
719 
720  dataCommunicator.setSend(rank, list.size() * dataSize);
721  }
722 
723  // Communicate the sign among the partitions
724  while (true) {
725  // If all partitions have propgate the sign on all cells we can exit
726  bool locallyComplete = true;
727  for (int region = 0; region < nRegions; ++region) {
728  if (regionType[region] == REGION_UNKNOWN) {
729  locallyComplete = false;
730  break;
731  }
732  }
733 
734  bool globallyComplete;
735  MPI_Allreduce(&locallyComplete, &globallyComplete, 1, MPI_C_BOOL, MPI_LAND, m_kernelPtr->getCommunicator());
736  if (globallyComplete) {
737  return;
738  }
739 
740  // Start the receives
741  for (const auto entry : mesh.getGhostExchangeTargets()) {
742  const int rank = entry.first;
743  dataCommunicator.startRecv(rank);
744  }
745 
746  // Start the sends
747  for (const auto entry : mesh.getGhostExchangeSources()) {
748  const int rank = entry.first;
749  auto &list = entry.second;
750  SendBuffer &buffer = dataCommunicator.getSendBuffer(rank);
751 
752  for (long id : list) {
753  // Detect the region associated to the ghost
754  int region = -1;
755  for (int k = 0; k < nRegions; ++k) {
756  if (regionCellList[k].count(id) != 0) {
757  region = k;
758  break;
759  }
760  }
761 
762  // Detect the sign to communicate.
763  if (regionType[region] == REGION_UNKNOWN) {
764  sign = 0;
765  } else if (regionType[region] == REGION_EXTERNAL) {
766  sign = 1;
767  } else if (regionType[region] == REGION_INTERNAL) {
768  sign = -1;
769  } else {
770  sign = getSign(id);
771  }
772 
773  buffer << sign;
774  }
775 
776  dataCommunicator.startSend(rank);
777  }
778 
779  // Check if it is possible to detect the sign of an unkown region
780  std::vector<int> regionSign(nRegions, 0);
781  for (int region = 0; region < nRegions; ++region) {
782  if (regionType[region] == REGION_EXTERNAL) {
783  regionSign[region] = 1;
784  } else if (regionType[region] == REGION_INTERNAL) {
785  regionSign[region] = -1;
786  }
787  }
788 
789  int nCompletedRecvs = 0;
790  while (nCompletedRecvs < dataCommunicator.getRecvCount()) {
791  int rank = dataCommunicator.waitAnyRecv();
792  const auto &list = mesh.getGhostExchangeTargets(rank);
793  RecvBuffer &buffer = dataCommunicator.getRecvBuffer(rank);
794 
795 
796  for (long id : list) {
797  buffer >> sign;
798  if (sign == 0) {
799  continue;
800  }
801 
802  // Detect the region associated to the ghost
803  int region = -1;
804  for (int k = 0; k < nRegions; ++k) {
805  if (regionCellList[k].count(id) != 0) {
806  region = k;
807  break;
808  }
809  }
810 
811  // Assign the sign to the region
812  if (regionType[region] != REGION_UNKNOWN) {
813  continue;
814  } else if (regionSign[region] != 0) {
815  assert(sign == regionSign[region]);
816  continue;
817  }
818 
819  regionSign[region] = sign;
820  }
821 
822  ++nCompletedRecvs;
823  }
824 
825  dataCommunicator.waitAllSends();
826 
827  // Set the sign of the fixable unkown regions
828  for (int region = 0; region < nRegions; ++region) {
829  if (regionType[region] != REGION_UNKNOWN) {
830  continue;
831  }
832 
833  sign = regionSign[region];
834  if (sign == 0) {
835  continue;
836  }
837 
838  // Set the type of the region
839  if (sign > 0) {
840  regionType[region] = REGION_EXTERNAL;
841  } else if (sign < 0) {
842  regionType[region] = REGION_INTERNAL;
843  }
844 
845  // Assign the sign to the whole region
846  assignSign(sign, regionCellList[region]);
847  }
848  }
849 #else
850  // Check that the sign has been propagated into all regions
851  bool complete = true;
852  for (int region = 0; region < nRegions; ++region) {
853  if (regionType[region] == REGION_UNKNOWN) {
854  complete = false;
855  break;
856  }
857  }
858 
859  assert(complete);
860 #endif
861
