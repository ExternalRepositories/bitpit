141 LevelSetIntersectionStatus LevelSetObject::intersectSurface(const long &i) const{
142  if( !isInNarrowBand(i)){
143  return LevelSetIntersectionStatus::FALSE;
144  }
145 
146  double circumcircle = m_kernelPtr->computeCellCircumcircle(i) ;
147  if(std::abs(getLS(i)) > circumcircle){
148  return LevelSetIntersectionStatus::FALSE;
149  }
150 
151  double incircle = m_kernelPtr->computeCellIncircle(i) ;
152  if(std::abs(getLS(i)) <= incircle){
153  return LevelSetIntersectionStatus::TRUE;
154  }
155 
156  return LevelSetIntersectionStatus::CLOSE;
157 }

328 void LevelSetObject::communicate( std::unordered_map<int,std::vector<long>> &sendList, std::unordered_map<int,std::vector<long>> &recvList, std::vector<adaption::Info> const *mapper){
329 
330  if( assureMPI() ){
331 
332  MPI_Comm meshComm = m_kernelPtr->getCommunicator() ;
333 
334  DataCommunicator sizeCommunicator(meshComm) ;
335  DataCommunicator dataCommunicator(meshComm) ;
336 
337  int offset = 2*getId() ;
338  sizeCommunicator.setTag(offset+0) ;
339  dataCommunicator.setTag(offset+1) ;
340 
341  /* Start receiving of first communications which contain the sizes
342  * of the communications of the exchanged data
343  */
344  for (const auto entry : sendList) {
345  int rank = entry.first;
346 
347  sizeCommunicator.setRecv( rank, sizeof(long) ) ;
348  sizeCommunicator.startRecv(rank);
349  }
350 
351  /* Fill the databuffer with its content from the LevelSetObject
352  * base class and specific derived class. Fill the sizebuffer
353  * with the size of the databuffer. Start sending both buffers
354  */
355  for (const auto entry : sendList) {
356 
357  // Get the send buffer
358  int rank = entry.first;
359 
360  sizeCommunicator.setSend(rank, sizeof(long) ) ;
361  dataCommunicator.setSend(rank, 0 ) ;
362 
363  SendBuffer &sizeBuffer = sizeCommunicator.getSendBuffer(rank);
364  SendBuffer &dataBuffer = dataCommunicator.getSendBuffer(rank);
365 
366  // Write data in databuffer
367  writeCommunicationBuffer( entry.second, dataBuffer ) ;
368 
369  // Write size of databuffer in sizebuffer
370  sizeBuffer << dataBuffer.capacity() ;
371 
372  // Start sending
373  sizeCommunicator.startSend(rank);
374  dataCommunicator.startSend(rank);
375  }
376 
377 
378  /* Check which size communications have arrived. For those which are
379  * available read the size of the databuffer and start receiving.
380  * Proceed once all sizeBuffers have arrived.
381  */
382  int nCompletedRecvs(0);
383  long dataSize;
384 
385  while (nCompletedRecvs < sizeCommunicator.getRecvCount()) {
386  int rank = sizeCommunicator.waitAnyRecv();
387  RecvBuffer &sizeBuffer = sizeCommunicator.getRecvBuffer(rank);
388 
389  sizeBuffer >> dataSize ;
390 
391  dataCommunicator.setRecv(rank,dataSize) ;
392  dataCommunicator.startRecv(rank) ;
393 
394  ++nCompletedRecvs;
395  }
396 
397 
398  /* In case of mesh adaption, the ids of the moved items must be freed
399  * in order to avoid conflicts when ids are recycled
400  */
401  if( mapper!=NULL){
402  clearAfterMeshAdaption(*mapper);
403  }
404 
405  /* Check which data communications have arrived. For those which are
406  * available start reading the databuffer into the data structure of
407  * LevelSetObject and its derived classes.
408  */
409  nCompletedRecvs = 0;
410  while (nCompletedRecvs < dataCommunicator.getRecvCount()) {
411  int rank = dataCommunicator.waitAnyRecv();
412 
413  RecvBuffer &dataBuffer = dataCommunicator.getRecvBuffer(rank);
414  readCommunicationBuffer( recvList[rank], dataBuffer ) ;
415 
416  ++nCompletedRecvs;
417  }
418 
419 
420  sizeCommunicator.waitAllSends();
421  sizeCommunicator.finalize() ;
422 
423  dataCommunicator.waitAllSends();
424  dataCommunicator.finalize() ;
425 
426  }
427 }
