64 long IndexGenerator::generateId()
65 {
66  // If the trash is empty generate a new id otherwise recycle the first id
67  // in the trash.
68  if (m_trash.empty()) {
69  if (m_highest == NULL_ID) {
70  m_highest = 0;
71  } else {
72  assert(m_highest < std::numeric_limits<long>::max());
73  ++m_highest;
74  }
75  m_latest = m_highest;
76  } else {
77  m_latest = m_trash.front();
78  m_trash.pop_front();
79  }
80 
81  return m_latest;
82 }

109 bool IndexGenerator::isIdAssigned(long id)
110 {
111  // If the latest id is valid, it is assigned by definition
112  if (m_latest >= 0 && id == m_latest) {
113  return true;
114  }
115 
116  // Ids past the highest one are not assigned
117  if (id > m_highest) {
118  return false;
119  }
120 
121  // The id is assigned only if is not in the trash
122  for (long trashedId : m_trash) {
123  if (trashedId == id) {
124  return false;
125  }
126  }
127 
128  return true;
129 }

136 void IndexGenerator::setAssignedId(long id)
137 {
138  // If the id is past the highest assigned id we need to trash all the ids
139  // from the highest assigned to this one (the generator only handles
140  // contiguous ids), otherwise look for the id in the trash and, if found,
141  // recycle it (if the id is not in the trash, this means it was already
142  // an assigned id).
143  if (id > m_highest) {
144  for (long wasteId = m_highest + 1; wasteId < id; ++wasteId) {
145  trashId(wasteId);
146  }
147 
148  m_highest = id;
149  } else {
150  for (auto trashItr = m_trash.begin(); trashItr != m_trash.end(); ++trashItr) {
151  long trashedId = *trashItr;
152  if (trashedId == id) {
153  m_trash.erase(trashItr);
154  break;
155  }
156  }
157  }
158 
159  // This is the latest assigned id
160  m_latest = id;
161 }

170 void IndexGenerator::trashId(const long &id)
171 {
172  m_trash.push_back(id);
173 
174  // We only keep track of the latest assigned id, if we trash that id we
175  // have no information of the previous assigned id.
176  if (id == m_latest) {
177  m_latest = NULL_ID;
178  }
179 }
