91 double LevelSetKernel::computeSizeNarrowBandFromLS( LevelSetObject *visitor, const bool &signd ){
92 
93  // We need to consider only the cells with a levelset value less than
94  // local narrow band (ie. size of the narrowband evalauted using the
95  // cell).
96  double newRSearch = 0.;
97  int factor ;
98 
99  for (auto &cell : getMesh()->getCells() ) {
100  // Discard cells outside the narrow band
101  long id = cell.getId() ;
102  if( !visitor->isInNarrowBand(id) ) {
103  continue;
104  }
105 
106  // Evaluate local search radius
107  std::array<double,3> myCenter = computeCellCentroid(id) ;
108 
109  const long* neighbours = cell.getAdjacencies() ;
110  int N = cell.getAdjacencyCount() ;
111 
112  double localRSearch = 0. ;
113  for(int n=0; n<N; ++n){
114  long neighId = neighbours[n] ;
115  if (neighId < 0) {
116  continue;
117  }
118 
119  if( visitor->isInNarrowBand(neighId)){
120  factor = (int) signd * visitor->getSign(id) + (int) (!signd) ;
121  std::array<double,3> diff = computeCellCentroid(neighId) - myCenter ;
122  if( factor *dotProduct(diff, visitor->getGradient(id)) < 0){
123  localRSearch = std::max( localRSearch, norm2(diff) );
124  }
125  }
126  }
127 
128  // Discard cells with a levelset greater than the local narrow band
129  if ( std::abs( visitor->getLS(id) ) > (localRSearch + m_mesh->getTol()) ) {
130  continue;
131  }
132 
133  // Update the levelset
134  newRSearch = std::max( localRSearch, newRSearch );
135  }
136 
137 # if BITPIT_ENABLE_MPI
138  if( assureMPI() ) {
139  double reducedRSearch ;
140  MPI_Allreduce( &newRSearch, &reducedRSearch, 1, MPI_DOUBLE, MPI_MAX, m_commMPI );
141  newRSearch = reducedRSearch ;
142  }
143 #endif
144 
145  return newRSearch;
146 
147 }

161 void LevelSetKernel::updateGeometryCache( const std::vector<adaption::Info> &mapper ) {
162 
163  // If there are no cells in the mesh we can just delete all the cache
164  if ( m_mesh->getCellCount() == 0) {
165  clearGeometryCache();
166  return;
167  }
168 
169  // Remove the previous cells from the cache
170  for ( auto & map : mapper ){
171  if( map.entity != adaption::Entity::ENTITY_CELL ){
172  continue;
173  }
174 
175  for ( auto & previousId : map.previous){
176  auto centroidItr = m_cellCentroids.find( previousId ) ;
177  if ( centroidItr == m_cellCentroids.end() ) {
178  continue ;
179  }
180 
181  m_cellCentroids.erase( previousId ) ;
182  }
183  }
184 
185 }

212 double LevelSetKernel::computeCellIncircle( long id ) {
213 
214  VolumeKernel *patch = getMesh();
215  Cell &cell = patch->getCell(id);
216 
217  const long* interfaceIds = cell.getInterfaces();
218  int interfaceCount = cell.getInterfaceCount();
219 
220  std::array<double,3> cellCenter = computeCellCentroid(id);
221 
222  double radius = std::numeric_limits<double>::max() ;
223  for (int k = 0; k < interfaceCount; ++k) {
224  long interfaceId = interfaceIds[k];
225  double r = norm2(cellCenter - patch->evalInterfaceCentroid(interfaceId));
226  radius = std::min(radius, r);
227  }
228 
229  int vertexCount = cell.getInterfaceCount();
230  for (int k = 0; k < vertexCount; ++k) {
231  long vertexId = cell.getVertex(k);
232  double r = norm2(cellCenter - patch->getVertexCoords(vertexId));
233  radius = std::min(radius, r);
234  }
235 
236  return radius;
237 
238 }

280 double LevelSetKernel::isCellInsideBoundingBox( long id, std::array<double, 3> minPoint, std::array<double, 3> maxPoint ){
281 
282  const Cell &cell = m_mesh->getCell(id);
283  double tolerance = m_mesh->getTol();
284 
285  std::array<double, 3> cellMinPoint;
286  std::array<double, 3> cellMaxPoint;
287 
288  cellMinPoint.fill( std::numeric_limits<double>::max() ) ;
289  cellMaxPoint.fill( - std::numeric_limits<double>::max() ) ;
290 
291  int nVertices = cell.getVertexCount();
292  for (int i = 0; i < nVertices; ++i) {
293  long vertexId = cell.getVertex(i);
294  std::array<double, 3> vertexCoords = m_mesh->getVertexCoords(vertexId);
295  for (int d = 0; d < 3; ++d) {
296  cellMinPoint[d] = std::min( vertexCoords[d] - tolerance, cellMinPoint[d]) ;
297  cellMaxPoint[d] = std::max( vertexCoords[d] + tolerance, cellMaxPoint[d]) ;
298  }
299  }
300 
301  return CGElem::intersectBoxBox(minPoint, maxPoint, cellMinPoint, cellMaxPoint);
302 }
